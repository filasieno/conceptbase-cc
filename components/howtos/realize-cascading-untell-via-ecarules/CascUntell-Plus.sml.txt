* ECArules for cascading UNTELL; additionally it allows to overwrite old attribute values.
* We recommend the CBserver option -s 1 when using the ECArules below.
* This will prevent undesired deletions of system objects.
* Requires ConceptBase 8.2.07 releases 2022-05-06 or later.
*
*}

{* *********************************************************************
   The cascade rules are react to UNTELL operations. They allow a
   transaction to only untell certain definitions. Attached attributes,
   instantiations and specializations are then  removed by the
   cascadeUntell* rules
   *********************************************************************
*}
{* These cascade rules will delete links to/from a deleted object *}
{* Hence, instead of following all dependencies, these            *}
{* rules will do most of the work.                                *}

ECArule cascadeUntellInClass with
  ecarule
    r1: $ x/Individual c/Proposition
         ON Untell (x in Individual)
         IF In_s(x,c)
         DO Untell (x in c)
        $
end

ECArule cascadeUntellClassOf with
  ecarule
    r1: $ x/Individual c/Proposition
         ON Untell (c in Individual)
         IF In_s(x,c)
         DO Untell (x in c)
        $
end

ECArule cascadeUntellInattributeeCategory with
  ecarule
    r1: $ a/attributee ac/attributee
         ON Untell (a in attributee)
         IF In_s(a,ac)
         DO Untell (a in ac)
        $
end

ECArule cascadeUntellattributeeCategoryOf with
  ecarule
    r1: $ a/attributee ac/attributee
         ON Untell (ac in attributee)
         IF In_s(a,ac) 
         DO Untell (a in ac)
        $
end

ECArule cascadeUntellToAttr with
  ecarule
    r1: $ x/Individual a/attributee
         ON Untell (x in Individual)
         IF To(a,x) and (a in attributee)
         DO Untell (a in attributee)
        $
end

ECArule cascadeUntellFromAttr with
  ecarule
    r1: $ x/Individual a/attributee
         ON Untell (x in Individual)
         IF From(a,x) and (a in attributee)
         DO Untell (a in attributee)
        $
end

ECArule cascadeUntellHasSuper with
  ecarule
    r1: $ c/Individual sc/Individual
         ON Untell (c in Individual)
         IF Isa_e(c,sc)
         DO Untell (c isA sc)
        $
end

ECArule cascadeUntellHasSub with
  ecarule
    r1: $ c/Individual sc/Individual
         ON Untell (sc in Individual)
         IF Isa_e(c,sc)
         DO Untell (c isA sc)
        $
end

ECArule cascadeUntellHasSuperAC with
  ecarule
    r1: $ c/attributee sc/attributee
         ON Untell (c in attributee)
         IF Isa_e(c,sc)
         DO Untell (c isA sc)
        $
end

ECArule cascadeUntellHasSubAC with
  ecarule
    r1: $ c/attributee sc/attributee
         ON Untell (sc in attributee)
         IF Isa_e(c,sc)
         DO Untell (c isA sc)
        $
end



{* *********************************************************************
   The overwrite rules are react to TELL operations. If a new attribute
   is told that has the same label as an old attribute of object x, then
   the old attribute is removed (UNTELL) by the ECArules. The first
   removed the instantiation of the old attribute to any attribute 
   category ac. The second removed the attribute itself.
   *********************************************************************
*}

{* this rule removes the attribute categories of an attribute a that needs to be removed 
   because a new attribute a1 is told that has the same label
*}
ECArule overwriteattributee1 with
  ecarule
    r1: $ a/attributee a1/attributee ac/attributee x/Proposition y/Proposition y1/Proposition n/Label 
         ON Tell (a1 in attributee)
         IF `P(a1,x,n,y1) and P(a,x,n,y) and In_s(a,ac)
         DO Untell (a in ac)
        $
end

{* the second rule removed the attribute a itself *}
ECArule overwriteattributee2 with
  ecarule
    r1: $ a/attributee a1/attributee ac/attributee x/Proposition y/Proposition y1/Proposition n/Label 
         ON Tell (a1 in attributee)
         IF `P(a1,x,n,y1) and P(a,x,n,y)
         DO Untell (a in attributee)
        $
end
