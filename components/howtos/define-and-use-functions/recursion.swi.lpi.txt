/* Some recursive numeric functions functions */
/* We basically call the SWI-Prolog implementation */

compute_pfib(_res,_n,_) :-
        cbserver:arg2val(_n,_nv),
        integer(_nv),
        _nv >= 0,
        lpi_fib(_vres,_nv),
        cbserver:val2arg(_vres,_res).


lpi_fib(0,0).
lpi_fib(1,1).
lpi_fib(_res,_n) :-
  _n > 1,
  _n1 is _n - 1,
  _n2 is _n - 2,
  lpi_fib(_r1,_n1),
  lpi_fib(_r2,_n2),
  _res is _r1 + _r2.
  


/* define the signature of 'pfib' to ConceptBase */
tell: 'pfib in Function isA Integer with parameter n: Integer end'.





compute_ack(_res,_m,_,_n,_) :-
        cbserver:arg2val(_m,_mv),
        cbserver:arg2val(_n,_nv),
        integer(_mv),
        integer(_nv),
        _mv >= 0, _nv >= 0,
        lpi_ack(_vres,_mv,_nv),
        cbserver:val2arg(_vres,_res).

/* Ackermann function in Prolog */
lpi_ack(_r,0,_n) :- _r is _n+1.
lpi_ack(_r,_m,0) :- _m > 0, _m1 is _m - 1, lpi_ack(_r,_m1,1).
lpi_ack(_r,_m,_n) :-
   _m > 0, _n > 0,
   _m1 is _m - 1,
   _n1 is _n - 1,
   lpi_ack(_r1,_m,_n1),
   lpi_ack(_r,_m1,_r1).



  


/* define the signature of 'ack' to ConceptBase */
tell: 'ack in Function isA Integer with parameter m: Integer; n: Integer end'.
