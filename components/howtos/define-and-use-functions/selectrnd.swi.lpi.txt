/* select a random instance of a class */
/* M. Jeusfeld, 23-Apr-2008 (24-Apr-2008) */

compute_selectrnd(_res,_class,_c1) :-
        nonvar(_class),                                  /* could be another query call        */
        cbserver:askAll(_x,[In(_x,_class)],_set),        /* _set holds all instances of _class */
        length(_set,_len),                               /* maybe 0 if there is no instance    */
        _len > 0,                                        /* _set must be non-empty             */
        _rnd is random(_len),                            /* range 0 <= _rnd < _len             */
        'GeneralUtilities':nth0(_rnd,_set,_res),         /* first element has index 0          */
        !.


/* define the signature of 'selectrnd' to ConceptBase */
tell: 'selectrnd in Function isA Proposition with parameter class: Proposition end'.
