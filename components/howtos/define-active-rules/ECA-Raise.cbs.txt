startServer -u nonpersistent -port 4411 -t high


tell "

Account with
  attribute
    balance: Real
end

{* This query is for calling the transferRule to transfer the amount between the accounts *}
RegularTransfer in GenericQueryClass isA YesClass with
  parameter
    amount: Real;
    fromaccount: Account;
    toaccount: Account
end 

{* This query is for calling transferRule_shared to transfer the amount between the accounts *}
SharedCostTransfer in GenericQueryClass isA YesClass with
  parameter
    amount: Real;
    fromaccount: Account;
    toaccount: Account
end 

{* This query is for taxing a given account via the taxRule *}
TaxAccount in GenericQueryClass isA YesClass with
  parameter
    account: Account;
    amount: Real;
    taxrate: Real
end 


{* in case of a regular transfer, we tax the from account with 4.5 % of the amount *}
ECArule transferRule with
  mode m: Deferred
  rejectMsg rm:
   \"Account balance insufficient\"
  ecarule
        er : $  acc1,acc2/Account tr/RegularTransfer m,b1,b2,nb1,nb2/Real
        ON Ask RegularTransfer[m,acc1,acc2] 
        IF
           `(acc1 balance b1) and
           `(acc2 balance b2) and
           (nb1 = b1 - m) and
           (nb2 = b2 + m) and
           (nb1 >= 0)
        DO Retell (acc1 balance nb1) ,
           Retell (acc2 balance nb2) ,
           Raise TaxAccount[acc1,m,0.045]
        ELSE
           reject
        $
end


{* in case of a shared cost transfer, we tax each account with 2.5 % of the amount *}
ECArule transferRule_shared with
  mode m: Deferred
  rejectMsg rm:
   \"Account balance insufficient\"
  ecarule
        er : $  acc1,acc2/Account tr/SharedCostTransfer m,b1,b2,nb1,nb2/Real
        ON Ask SharedCostTransfer[m,acc1,acc2] 
        IF
           `(acc1 balance b1) and
           `(acc2 balance b2) and
           (nb1 = b1 - m) and
           (nb2 = b2 + m) and
           (nb1 >= 0)
        DO Retell (acc1 balance nb1) ,
           Retell (acc2 balance nb2) ,
           Raise TaxAccount[acc1,m,0.025], 
           Raise TaxAccount[acc2,m,0.025]
        ELSE
           reject
        $
end

{* deduce r*m from account acc *}
ECArule taxRule with
  mode m: Deferred
  ecarule
        er : $  acc/Account m,b,nb,r/Real
        ON Ask TaxAccount[acc,m,r] 
        IF
           `(acc balance b) and
           (nb = b - r*m) 
        DO Retell (acc balance nb)
        $
end

"

tell "
PetersAccount in Account with
  balance b: 500.0
end 

MarysAccount in Account with
  balance b: 1.0
end
"

ask "RegularTransfer[250.0/amount,PetersAccount/fromaccount,MarysAccount/toaccount]" OBJNAMES  FRAME  Now
ask "SharedCostTransfer[150.0/amount,PetersAccount/fromaccount,MarysAccount/toaccount]" OBJNAMES  FRAME  Now



ask "get_object[PetersAccount/objname]" OBJNAMES  FRAME  Now

ask "get_object[MarysAccount/objname]" OBJNAMES  FRAME  Now
