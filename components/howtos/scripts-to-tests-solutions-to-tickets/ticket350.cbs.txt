# Test objects referring to their attributes in their defining frame
# Call like
#    cbshell ticket350.cbs




startServer -u nonpersistent -t high 


tell "
OBJECT in Class with
  attribute
    link: OBJECT
  single 
    iLink: OBJECT!link
end
"
# this frame should be compiled


tell "
OBJECT1 with
  attribute
    link: OBJECT1;   
    iLink: OBJECT1!undefined
end
"
# this frame should be rejected


tell "
OBJECT2 with
  attribute
    iLink: OBJECT2!link;
    link: OBJECT2
end
"
# this frame is also rejected since OBJECT2!link is used before defined

ask get_object[OBJECT/objname] OBJNAMES  FRAME  Now
# should return the first frame 

ask listModule
