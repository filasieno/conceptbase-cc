startServer -u nonpersistent -t high

tell "
OBJECT with
  attribute
    ISA: OBJECT;
    ISA_t: OBJECT
end

ClosureRules in Class with
  rule


    c3: $ forall x,y/OBJECT (x ISA y) ==> (x ISA_t y) $;

    t3: $ forall x,y,z/OBJECT (x ISA z) and (z ISA_t y) ==> (x ISA_t y) $
end

DDI_SyntacticRestrictions in Class with
  constraint

    {* acyclicity of isa *}
    a1: $ forall x,c/OBJECT (x ISA_t c) ==> (x <> c) $
end


"



tell "
vx in OBJECT with
  ISA c1: vx
end

"
