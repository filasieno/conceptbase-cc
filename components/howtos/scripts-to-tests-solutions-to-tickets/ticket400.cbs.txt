# Allow emptiness constraints for query classes
#

cbserver -t low
# connect

tell "
Employee in Class with  
  attribute
    salary : Integer
end 
"




tell "
mary in Employee with  
  salary
    s : 1000
end 
"

# This is a query class that we want to make subject to an integrity constraint.
# To do do, it also has to be an instance of MSFOLrule (rules that are subject
# to incremental integrity checking)

tell "
LowSalaryEmployee in QueryClass,MSFOLrule isA Employee with  
  constraint
    c1 : $ exists s/Integer (this salary s) and (s < 500) $
end 
"



# This is a class whose instances must all have an empty extension
tell "
EmptyClass in Class with  
  constraint
    isEmpty : $ forall ec/EmptyClass not exists x/Proposition (x in ec) $
end 
"

# Now, we demand that LowSalaryEmployee must have an empty extension
tell "LowSalaryEmployee in EmptyClass end"

ask get_object[LowSalaryEmployee/objname] OBJNAMES FRAME Now

# the generated constraint is attached to the superclass of the query because
# query classes have a different meaning of 'constraint'
ask get_object[Employee/objname] OBJNAMES FRAME Now


# This update should be rejected because bill would be in the extension
# of LowSalaryEmployee
tell "
bill in Employee with  
  salary
    s : 100
end "



# Now, do a similar thing for a non-emptiness constraint

tell "
NonEmptyClass in Class with  
  constraint
    isNotEmpty : $ forall nec/NonEmptyClass exists x/Proposition (x in nec) $
end 
"

tell "
HighSalaryEmployee in QueryClass,MSFOLrule isA Employee with  
  constraint
    c1 : $ exists s/Integer (this salary s) and (s >= 1000) $
end 
"

tell "HighSalaryEmployee in NonEmptyClass end"
#note that mary is an instance of HighSalaryEmployee


untell "
mary with  
  salary
    s : 1000
end 
"
# this should fail because there would be no other employee left with a high salary
