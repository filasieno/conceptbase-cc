GraphVizType end

Boxnode in GraphVizType,Class with
  rule
    r1: $ forall n/Deliverable (n in Boxnode) $
end

Rboxnode in GraphVizType,Class with
  rule
    r1: $ forall n/Activity :(n in Activity): and 
          not (n in Diamondnode) and not (n in Barnode) and not (n in Subgraphnode) ==> (n in Rboxnode) $
end

Diamondnode in GraphVizType,Class with
  rule
    r1: $ forall n/DecisionPoint  (n in Diamondnode) $
end

Barnode in GraphVizType,Class with
  rule
    r1: $ forall n/Activity (n in ParallelBranch) or (n in ParallelJoin)  ==> (n in Barnode) $
end

Subgraphnode in GraphVizType,Class with
  rule
    r1: $ forall n/ComplexActivity (n in Subgraphnode) $
end

Link end

Controllink in GraphVizType,Class isA Link with
  rule
    r1: $ forall l/ActivityNode!next (l in Controllink) $
end

Crosslink in GraphVizType,Class isA Link with
  rule
    r1: $ forall l/Activity!produces (l in Crosslink) $
end

Containslink in GraphVizType,Class isA Link with
  rule
    r1: $ forall l/OpenConcept!contains (l in Containslink) $
end

Regularlink in GraphVizType,Class isA Link with
  rule
    r1: $ forall l/NodeOrLink!connectedTo not (l in OpenConcept!contains) ==> (l in Regularlink) $
end


PDDContainer with
  attribute
    hasElem: PDDElement
end

PDDElement end

ActivityNode isA PDDElement end
ActivityNode!next isA PDDElement end
Deliverable isA PDDElement end
OpenConcept isA PDDElement end
OpenConcept!contains isA PDDElement end


Function LinkLabel isA Label with
  parameter
     li: Link
  constraint
     cfn: $ (li in DecisionPoint!choice) and Label(li,this)
            or (not (li in DecisionPoint!choice)) and (this ="")
          $
end


GenericQueryClass ShowDeliverable isA Deliverable with
  required,parameter
     del: Deliverable
  constraint
     c1: $ (del = this)  $
end


Deliverable in Class isA PDDContainer with
  rule
     r1: $ forall d/Deliverable (d hasElem d) $;
     r2: $ forall d1/OpenConcept d2/Deliverable e/PDDElement (d1 contains d2) and (d2 hasElem e) ==> (d1 hasElem e) $;
     r3: $ forall d/OpenConcept acon/OpenConcept!contains From(acon,d) ==> (d hasElem acon) $;
     r4: $ forall d1,d2/Deliverable (d1 connectedTo d2) ==> (d1 hasElem d2) $;
     r5: $ forall d/Deliverable acon/NodeOrLink!connectedTo From(acon,d) ==> (d hasElem acon) $
end



DELIVERABLE_FORMAT in AnswerFormat with
  forQuery q: ShowDeliverable
  pattern p:
"{ASKquery(ShowElements[{this}/container,Boxnode/type],BOXNODE_FORMAT)}
{ASKquery(ShowElements[{this}/container,Regularlink/type],REGULARLINK_FORMAT)}
{ASKquery(ShowElements[{this}/container,Containslink/type],CONTAINSLINK_FORMAT)}
"
end



GenericQueryClass ShowPDD isA Phase with
  required,parameter
     process: Phase;
     deliverable: Deliverable
  constraint
     c1: $ (process = this)  $
end


PDD_FORMAT in AnswerFormat with
  forQuery q: ShowPDD
  pattern p:
"# This is a Process-Data-Diagram used for relating processes to data
# See also http://en.wikipedia.org/wiki/Process-data_diagram
# Graphviz code generated by ConceptBase {cb_version} at {transactiontime} 
# Database: {database}
# Process this file by Graphviz, e.g. 
#    dot -Tpng thisfile.dot > thisfile.png

digraph {ALPHANUM({process})}__{ALPHANUM({deliverable})} \{
{ASKquery(ShowDeliverable[{deliverable}/del],DELIVERABLE_FORMAT)}

subgraph cluster_0 \{
{ASKquery(ShowPhase[{process}/phase],PHASE_FORMAT)}
{ASKquery(ShowOrphans[{process}/phase,{deliverable}/del],ORPHAN_FORMAT)}
{ASKquery(ShowCrosslinks[{process}/phase],CROSSLINK_FORMAT)}
label=\"{ALPHANUM({this})}\";color=gray;
\}


#overlap=false;
fontsize=12;
concentrate=true;
ranksep=0.32;
#splines=false;
label=\"PDD {ALPHANUM({process})} from database {database} ({transactiontime} UTC)\"
\}
"
end

{* Orphans are deliverables that are produced by some activity of phase but *}
{* that are not part of the main deliverable del.                           *}

GenericQueryClass ShowOrphans isA Deliverable with
  required,parameter
     phase: Phase;
     del: Deliverable
  constraint
     c1: $ exists a/Activity (phase subactivity a) and (a produces this) and 
           not (del hasElem this) $
end


GenericQueryClass ShowCrosslinks isA Activity!produces with
  required,parameter
     phase: Phase
  constraint
     c1: $ exists a/Activity (phase subactivity a) and Ai(a,produces,this)  $
end




GenericQueryClass ShowPhase isA Phase with
  required,parameter
     phase: Phase
  constraint
     c1: $ (phase = this)  $
end



Phase in Class isA PDDContainer with
  rule
    rs1: $ forall ph/Phase a/Activity (ph activity a) ==> (ph hasElem a) $;
    rs2: $ forall ph/Phase a/Activity anext/ActivityNode!next (ph activity a) and From(anext,a) ==> (ph hasElem anext) $
end



GenericQueryClass ShowElements isA PDDContainer with
  required,parameter
     container: PDDContainer;
     type: GraphVizType
  computed_attribute
     elem: PDDElement
  constraint
     c1: $ (container = this)  and
           (this hasElem elem) and
           (elem in type) $
end


PHASE_FORMAT in AnswerFormat with
  forQuery q: ShowPhase
  pattern p:
"{ASKquery(ShowElements[{this}/container,Subgraphnode/type],SUBGRAPH_FORMAT)}
{ASKquery(ShowElements[{this}/container,Rboxnode/type],RBOXNODE_FORMAT)}
{ASKquery(ShowElements[{this}/container,Boxnode/type],BOXNODE_FORMAT)}
{ASKquery(ShowElements[{this}/container,Diamondnode/type],DIAMONDNODE_FORMAT)}
{ASKquery(ShowElements[{this}/container,Barnode/type],BARNODE_FORMAT)}
{ASKquery(ShowElements[{this}/container,Controllink/type],PROPAGATEDLINK_FORMAT)}"
end


PHASECLUSTER_FORMAT in AnswerFormat with
  pattern p:
"subgraph cluster_{this.oid} \{
{ASKquery(ShowElements[{this}/container,Rboxnode/type],RBOXNODE_FORMAT)}
{ASKquery(ShowElements[{this}/container,Diamondnode/type],DIAMONDNODE_FORMAT)}
{ASKquery(ShowElements[{this}/container,Barnode/type],BARNODE_FORMAT)}
{ASKquery(ShowElements[{this}/container,Controllink/type],CONTROLLINK_FORMAT)}
style=filled;color=gainsboro;label=\"{this}\";labeljust=l;
\}

{ASKquery(ShowElements[{this}/container,Boxnode/type],BOXNODE_FORMAT)}
{ASKquery(ShowElements[{this}/container,Containslink/type],CONTAINSLINK_FORMAT)}
"
end

end

end

DIAMONDNODE_FORMAT in AnswerFormat with

end

BARNODE_FORMAT in AnswerFormat with
pattern p: "node [shape=box,style=filled,fillcolor=black,label=\"\",width=1.0,height=0.1]; {Foreach( ({this.elem}), (d), {d};)}"

CONTROLLINK_FORMAT in AnswerFormat with
pattern p: "{Foreach( ({this.elem}),(l),{From({l})}->{To({l})} [weight=5,color=black,len=1.21,label={ASKquery(LinkLabel[{l}/li],LABEL)}];\\n)}"
end

CONTAINSLINK_FORMAT in AnswerFormat with
pattern p: "{Foreach( ({this.elem}),(l),{From({l})}->{To({l})} [arrowtail=odiamond,arrowhead=none,len=1.20,label=\"\"];\\n)}"
end

REGULARLINK_FORMAT in AnswerFormat with
pattern p: "{Foreach( ({this.elem}),(l),{From({l})}->{To({l})} [dir=none,len=1.20,label=\"\"];\\n)}"
end

SUBGRAPH_FORMAT in AnswerFormat with
pattern p: "{Foreach( ({this.elem}),(l),{ASKquery(ShowPhase[{l}/phase],PHASECLUSTER_FORMAT)})}"
end

PROPAGATEDLINK_FORMAT in AnswerFormat with
pattern p: "{Foreach( ({this.elem}),(l),{ASKquery(EndNode[{From({l})}/diagram],LABEL)}->{ASKquery(StartNode[{To({l})}/diagram],LABEL)} [color=black,len=1.22,label={ASKquery(LinkLabel[{l}/li],LABEL)}];\\n)}"
end

CROSSLINK_FORMAT in AnswerFormat with
pattern p: "{From({this})}->{To({this})} [arrowhead=vee,color=black,len=1.20,label=\"\",style=dashed];\\n"
end

ORPHAN_FORMAT in AnswerFormat with
pattern p: "node [shape=box,style=filled,fillcolor=whitesmoke]; {this};"
end
