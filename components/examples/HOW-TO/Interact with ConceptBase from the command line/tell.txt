#!/usr/bin/cbshell
# Syntax: tell <port> <frame>
# Author: Manfred Jeusfeld 
# Created: 2012-06-14
# -----------------------------------------------------------------------
# Tell the <frame> to the CBserver running on localhost and port number <port>
#
# Requires ConceptBase 7.4.02
# The cbshell script shall be copied to /usr/bin in order to use this script
# You need to start a CBserver on a certain port number, e.g. 4001, prior
# to starting this script.
#
# Example:
#   tell 4001 "Employee in Class end"




enrollMe localhost $1
# the port number is supplied as parameter 1 of the script

tell "$2"
showAnswer
# if you do not want to see the error messages, then put the next line in comments
getErrorMessages
