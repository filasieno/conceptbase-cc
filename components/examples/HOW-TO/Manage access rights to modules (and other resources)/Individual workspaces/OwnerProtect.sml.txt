{*
* File: OwnerProtect.sml
* Author: Manfred Jeusfeld (18-Sep-2007, 4-Feb-2014)
* ----------------------------------------------------------------------------
* This file protects the workspace owned by a specific user.
* Owners can do read/write, others can do nothing.
*
* See ConceptBase User Manual, appendix F, for more details.
* Note that you need to start the CBserver with flag -s 2 (securityLevel 2).
* Otherwise, the CBserver ignores any access policy!
*
* This model requires ConceptBase 7.3 released 8-Dec-2011 or later.
*
*}


{* The default home module 'oHome' will host all the individual workspaces *}

oHome in AutoHomeModule end


{* The subsequent definitions specify the access policy of individual *}
{* workspaces. We first add the attributes permitted_read,            *}
{* permitted_write and owner_resource to the CB_User class.           *}

CB_User with 
  attribute
    permitted_read: Resource;
    permitted_write: Resource;
    owner_resource: Resource
end

{* Then, we define that the owner of a resource (module) can write to it *}
{* and that anyone who can write can also read a resource.               *}

CB_User in Class with
  rule
   rr1: $ forall p/Resource u/CB_User (u owner_resource p) ==> (u permitted_write p) $;
   rr2: $ forall p/Resource u/CB_User (u permitted_write p) ==> (u permitted_read p) $
end

UNTELL in CB_WriteOperation end  {* some additional CB_Operation beyond the predefined ASK,TELL *}


{* CB_Permitted implements our policy. Modules without an owner shall *}
{* shall be ragarded as unprotected.                                  *}

GenericQueryClass CB_Permitted isA CB_User with
  parameter
    user: CB_User;
    res: Resource;
    op: CB_Operation
  constraint
    cperm: $ (
              ( not exists u/CB_User (u owner_resource ~res) ) or
              ( (~op in CB_ReadOperation) and (~user permitted_read ~res) ) or
              ( (~op in CB_WriteOperation) and (~user permitted_write ~res) )
             )
              and UNIFIES(~user,~this) $
end

{* The subsequent active rule sets the owner of a new module to be *}
{* the creator of the module. This rule is triggered every time a  *}
{* new module is told. Builtin predicate currentUser returns the   *}
{* CB_User who has initiated the current transaction, i.e. the     *}
{* transaction that contains the telling of the new module.        *}
{* We prevent the ECA rule from setting ownership to the module    *}
{* 'System'.                                                       *}                             

ECArule SetOwnership with
  mode m: Deferred
  ecarule
        er : $  m/Module u/CB_User
        ON Tell( (m in Module) )
        IF not (m = System) and not (m = oHome)
        DO CALL(currentUser(u)),
           Tell( (u owner_resource m) )
        $
end

{* We protect the home module 'oHome' separately.            *}
{* Actually, we do not allow anybody to modify it.           *}

GenericQueryClass CB_PermittedoHome isA CB_User with
  parameter
      user: CB_User;
      res: Resource;
      op: CB_Operation
  constraint
    cperm: $ (~op in CB_ReadOperation) and UNIFIES(~user,~this) $
end
