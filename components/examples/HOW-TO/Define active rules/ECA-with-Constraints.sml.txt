* This ECA rule is affecting an integrity constraint.
* An update that violates the constraint is rejected.
* For the example below, update [3] will cause the 
* integrity violation.
*
* Requires ConceptBase 7.1 released after 2-Apr-2008.
*
}


{* The salary of employess must be below 1000 *}

Employee in Class with
  attribute
    salary: Integer;
    supervises: Employee
  constraint
    c1: $ forall e/Employee s/Integer (e salary s) ==> (s < 1000) $
end


{* This ECA rule will increase the salary of an employee by 100 if the *}
{* employee supervises another employee. Each new supervision will     *}
{* lead to another salary increase.                                    *}
{* The Retell first removes attribute matching (e1 salary x) and then  *}
{* performs a Tell of (e1 salary s2).                                  *}
{* We need to use the evaluate (e1 salary s1) on the new database      *}
{* because one can tell several supervises attributes in one           *}
{* transaction, leading to revisions of the salary attribute.          *}

ECArule IncreaseSalary with
  mode m: Deferred
  ecarule
        er : $  e1,e2/Employee s1,s2/Integer
        ON Tell (e1 supervises e2)
        IF `(e1 salary s1) and (s2=s1+100)
        DO Retell (e1 salary s2)
        $
end

{* Init the salary to 100 if employee e1 has not yet a salary *}

ECArule InitSalary with
  mode m: Deferred
  ecarule
        er : $  e1,e2/Employee s1/Integer
        ON Tell (e1 supervises e2)
        IF not (e1 salary s1) 
        DO Tell (e1 salary 100)
        $
end






{* Script to test the above ECA rule against the salary constraint: 

[1] set salary of bill to 850 and declare john and mary

bill in Employee with
  salary
    s1: 850
end
john in Employee end
mary in Employee end


[2] declare bill to supervise john

bill in Employee with
  supervises e1: john
end

==> triggers ECA rule IncreaseSalary to promote bill's salary to 950


[3] declare bill to supervise mary

bill in Employee with
  supervises e2: mary
end

==> triggers IncreaseSalary again to raise bill's salary to 1050; this causes
    an integrity violation and update [3] is rejected

*}
