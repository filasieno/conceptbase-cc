:- global CURRENTTIME/1 .

{* the answer format generator expects a substring to be a list
of atoms. Hence, we generate a list that contains just a single
atom which is the current time.
The format of the current time is a bit strange. It is used
here to demonstrate the call of external predicates. 
Knowledge of ConceptBase source code and Prolog
is required to use this feature.
*}

CURRENTTIME(_buf) :-
  currenttime$prologToUnixSUN4(_ct),
  prettyDateTime(_ct,_ta),
  appendBuffer$ExternalCodeLoader(_buf,_ta),
  !.


prettyDateTime(millisecond(_y,_mo,_d,_h,_mi,_s,_),_ta) :-
  pc_inttoatom$PrologCompatibility(_y,_yat),
  pc_inttoatom$PrologCompatibility(_mo,_moat),
  pc_inttoatom$PrologCompatibility(_d,_dat),
  pc_inttoatom$PrologCompatibility(_h,_hat),
  pretty2(_mi,_miat),
  pretty2(_s,_sat),
  pc_atomconcat$PrologCompatibility([_dat,'.',_moat,'.',_yat,' ',_hat,':',_miat,':',_sat,' GMT'],_ta).

pretty2(_x,_xat) :-
  _x >= 0,
  _x =< 9,
  !,
  pc_inttoatom$PrologCompatibility(_x,_xat1),
  pc_atomconcat$PrologCompatibility('0',_xat1,_xat).


pretty2(_x,_xat) :-
   pc_inttoatom$PrologCompatibility(_x,_xat).
