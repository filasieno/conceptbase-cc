# Check capabilities of ECArules to prescribe a certain order of actions
# Inspired by figure 4-10 of master thesis of L. Liu
#

startServer -u nonpersistent -port 4411 -t veryhigh

tell "

y1 end
y2 end

A in Class end
B in Class end
C in Class end
D in Class end
E in Class end

D1 in Class end
D2 in Class end



ECArule onA with
  mode m: Deferred
  ecarule
        er : $  x/Proposition
        ON TRANSACTIONAL Tell (x in A) 
        IF TRUE
        DO Tell (x in B),
           Tell (x in C)
        $
end

ECArule onB with
  mode m: Deferred
  ecarule
        er : $  x/Proposition
        ON Tell (x in B) 
        IF TRUE
        DO Tell (x in D)
        $
end

ECArule onC with
  mode m: Deferred
  ecarule
        er : $  x/Proposition
        ON Tell (x in C) 
        IF TRUE
        DO Tell (x in E)
        $
end

ECArule onD with
  mode m: Immediate
  ecarule
        er : $  x/Proposition
        ON Tell (x in D) 
        IF TRUE
        DO Tell (x in D1),
           Tell (x in D2)
        $
end

"


tell "
x1 in A end
x2 in A end
"

# Desired action sequence should separate actions for x1 and x2
#  Fire rule onA for event Tell (x1 in A)
#  DO Tell (x1 in B)
#  DO Tell (x1 in C)
#  DO Tell (x1 in D)
#  DO Tell (x1 in D1)
#  DO Tell (x1 in D2)
#  DO Tell (x1 in E)
#  Fire rule onA for event Tell (x2 in A)
#  DO Tell (x2 in B)
#  DO Tell (x2 in C)
#  DO Tell (x2 in D)
#  DO Tell (x2 in D1)
#  DO Tell (x2 in D2)
#  DO Tell (x2 in E)



# Deactivate onA for 2nd test
tell "
onA with
active act: FALSE
end
"


# 2nd test: use FOR-x-clause
tell "
ECArule onA2 with
  mode m: Deferred
  ecarule
        er : $  x/Proposition
        ON Tell (x in A) FOR x
        IF TRUE
        DO Tell (x in B),
           Tell (x in C)
        $
end
"

tell "
f1 in A end
f2 in A end
"

# Desired action sequence should separate actions for f11 and f2
# much like for x1,x2 before
