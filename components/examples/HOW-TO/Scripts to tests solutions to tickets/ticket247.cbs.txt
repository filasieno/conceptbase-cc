startServer -u nonpersistent -p 4411 -t high
tell "

Account with
  attribute
    balance: Integer
end

Transfer with
  attribute
    fromaccount: Account;
    amount: Integer
end
    A end

ECArule transferRule with
  mode m: Deferred
  ecarule
        er : $  acc1,acc2/Account tr/Transfer m,b1,b2,nb1,nb2/Integer a/Individual
        ON Tell (tr amount m) 
        IF `(tr fromaccount acc1) 
        DO CALL CreateIndividual(A,a),
           Tell (a in Account),
           Tell (a balance m)
        $
end
"

tell "
PetersAccount in Account with
  balance b: 500
end

"


# two transfers in one transction: should fail
tell "
transfer1 in Transfer with
   fromaccount a1: PetersAccount
   amount m: 450
end
"


ask "get_object[PetersAccount/objname]" OBJNAMES  FRAME  Now
ask "get_object[A_1/objname]" OBJNAMES  FRAME  Now
