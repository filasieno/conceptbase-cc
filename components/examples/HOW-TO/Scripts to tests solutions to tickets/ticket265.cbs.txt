startServer -u nonpersistent -p 4411 -t high
tell "
NodeOrLink with  
  attribute
    connectedTo : NodeOrLink
end 

Node isA NodeOrLink  
end 

NodeOrLink!connectedTo isA NodeOrLink  
end 


EntityType with
  attribute
    disjointTo: EntityType
end


ISA_Type in Node with  
  connectedTo
    super : EntityType
  connectedTo
    sub : EntityType
end 

EntityType in Node,Class with  
  attribute
    superType : EntityType
  rule
    r1 : $ forall I/ISA_Type e1,e2/EntityType (I super e1) and (I sub e2) ==> (e2 superType e1) $;
    r2 : $ forall e1,e2/EntityType x/VAR (x in e1) and (e1 superType e2) ==> (x in e2) $
end 

ISA_Type_disjoint in Node,Class isA ISA_Type  with
  rule
    dr1: $ forall e1,e2/EntityType ISAD/ISA_Type_disjoint
             (ISAD sub e1) and (ISAD sub e2) ==> (e1 disjointTo e2) $
end 


"

# This used to cause the crash of ticket 265
tell "
ISA_Type_disjoint in Class with  
  constraint
    disjC : $ forall x/VAR e1,e2/EntityType (e1 disjointTo e2) and (x in e1) and (x in e2)
                     ==> (e1 = e2) $
end 
"
