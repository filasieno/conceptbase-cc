# variant that uses a self-defined ISA 
tell "
    Proposition with
      attribute
         inViaConcrete: Proposition
    end
    CLASS in Class with
      attribute
        ISA: CLASS
      rule
        rISA1: $ forall c/CLASS (c ISA c) $;
        rISA2: $ forall c,d,e/CLASS :(c ISA d): and (d ISA e) ==> (c ISA e) $;
        rISA3: $ forall x/Proposition c,d/CLASS (x in c) and (c ISA d) ==> (x in d) $
    end    
"

tell "
    ABSTRACTCLASS isA CLASS end
    CONCRETECLASS in QueryClass,MSFOLrule isA CLASS with
      constraint
         concr: $ not (this in ABSTRACTCLASS) $  
    end
    ABSTRACTCLASS in Class with
      rule
         r1: $ forall x/VAR sc,ac/CLASS 
                  (sc ISA ac) and (ac in ABSTRACTCLASS) and (sc in CONCRETECLASS) and (x in sc) ==> (x inViaConcrete ac) $
      constraint
         noDirectInstance: $ forall ac/ABSTRACTCLASS x/Proposition (x in ac)  ==> (x inViaConcrete ac) $
    end
"
