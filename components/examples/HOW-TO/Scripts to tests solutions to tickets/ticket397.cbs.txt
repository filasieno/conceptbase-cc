cbserver -t high


tell '
Department in Class with
   attribute
     budget: Integer
end

Employee in Class with
   attribute
     salary: Integer;
     worksFor: Department
   rule
     r1: $ forall e/Employee s/Integer (e salary s) and (s < 1000) ==> (e gproperty/bgcolor "240,240,240") $;
     r2: $ forall e/Employee s/Integer (e salary s) and (s >= 1000) ==> (e gproperty/bgcolor "255,255,200") $;
     r3: $ forall e/Employee d/Department b/Integer (e worksFor d) and (d budget b) and (b > 20000) ==> 
                (e gproperty/linecolor "240,0,0") $;
     r4: $ forall e/Employee (exists d/Department (e worksFor d)) ==> (e gproperty/linewidth 3) $
end
'


tell '
Finance in Department with
  budget
    b: 25000
end

Production in Department with
  budget
    b: 15000
end


bill in Employee with
  worksFor d: Production
  salary s: 500
end

mary in Employee with
  worksFor d: Production
  salary s: 1500
end

anne in Employee with
  worksFor d: Finance
  salary s: 1700
end

john in Employee with
  worksFor d: Finance
  salary s: 700
end

sara in Employee end

'
