# EvenOdd again broken; only retuns the first value of Even
# Start: CBshell -f ticket286.cbs

startServer -u nonpersistent -port 4411 -t high


tell "
Int in Class isA Integer with
  rule
    axiom1: $ (0 in Int) $;
    axiom2: $ forall n/Int n1/Integer (n1 = n+1) and (n < 10)
                 ==> (n1 in Int) $
end
"



tell "
Individual Odd end
QueryClass Even isA Int with
  constraint
    c: $ (this = 0) or (exists y/Odd (this = y+1) ) $
end

QueryClass Odd isA Int with
  constraint
    c: $ exists y/Even (this = y+1)  $
end

GenericQueryClass isEven isA Boolean with
  parameter x: Int
  constraint ceven: $ (x in Even) and (this = TRUE) $
end

GenericQueryClass isInt isA Boolean with
  parameter x: Integer
  constraint ceven: $ (x in Int) and (this = TRUE) $
end
"
