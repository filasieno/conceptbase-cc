# This script tests the solution to ticket 212 (wrong behavior of Ai predicate)
#

startServer -u nonpersistent -t low 

# First tell a class Person with some instances
tell "Person with attribute friend: Person end
john in Person end
bill in Person end"

# constraint c1 uses the A predicate; the first tell should fail since
# the constraint c1 is not satisfied
tell "Person in Class with constraint c1: $ (john friend bill) $ end"

# tell that john has friend bill
tell "john with friend f1: bill end"

# Now, the constraint c1 is ok. 
tell "Person in Class with constraint c1: $ (john friend bill) $ end"

# The untell should fail due to constraint c1
untell "john with friend f1: bill end"

# The constraint c2 demands that john has an friends attribute john!f1
# This is true due to an earlier tell 
tell "Person in Class with constraint c2: $ Ai(john,friend,john!f1) $ end"

# So, the untell of john!f1 should fail due to constraint c2
untell "john with friend f1: bill end"

# Now let is attach an attribute f1 to bill; the attribute is not a 'friend' attribute yet
tell "bill with attribute f1: john end"

# Consequently, the constraint c3 should be ok
tell "Person in Class with constraint c3: $ not Ai(bill,friend,bill!f1) $ end"

# But assigning bill!f1 as a friend attribute will trigger a violation of c3
tell "bill with friend f1: john end"

# Another example to test Ai, this time via the meta formula specifying that an attribute is single-valued
tell "Class with constraint, attribute
singleConstraint :
          $ forall c,d/Proposition p/Proposition!single x,m/VAR
              P(p,c,m,d) and (x in c) ==>
                (
                  forall a1,a2/VAR
                    (a1 in p) and (a2 in p) and Ai(x,m,a1) and Ai(x,m,a2) ==>
                   (a1=a2)
                ) $
end
"

# The attribute testattr is defined as single-valued
tell "Class Test with
single
  testattr : Integer
end"

# This definition is fine since s1 is currently the only testattr
tell "bla3 in Test with
testattr s1 : 1
end"

# The following definition must be rejected because it defines a second testattr s2
tell "bla3 in Test with
testattr s2 : 2
end"
