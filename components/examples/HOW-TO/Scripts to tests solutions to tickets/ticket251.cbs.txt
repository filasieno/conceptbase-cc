startServer -u nonpersistent -port 4411 -t high


tell "

A end

C isA D end

D with
end

"



tell "
ECArule AtoC with
  mode m: ImmediateDeferred
  ecarule
        er : $  x/Proposition
        ON Tell (x in A) 
        IF FALSE
        DO noop 
        ELSE Tell (x in C) 
        $
end


ECArule forD with
  mode m: ImmediateDeferred
  ecarule
        er : $  x/Proposition
        ON Tell (x in D) 
        IF TRUE
        DO Tell (D attribute x)
        $
end

"


tell "
X1 in A end
"




ask "get_object[D/objname]" OBJNAMES  FRAME  Now
# should have X1  as attribute value
