# Test for detecting of reserve words as variable names;
# they should be rejected and an appropriate error message
# should be issued
# Start: CBshell -f ticket293.cbs
#



startServer -u nonpersistent -t high 

tell "
Employee in Class with
  attribute
    salary: Integer
end
"



# this one uses 'new' and does not compile
tell "
Employee in Class with
 constraint
    c_fail1: $ forall new/Employee s/Integer (new salary s) ==> (s > 100) $
end
"

# this one uses 'in' and does not compile
tell "
Employee in Class with
 constraint
    c_fail2: $ forall in/Employee s/Integer (new salary s) ==> (s > 100) $
end
"


# this one uses 'nw' and does compile
tell "
Employee in Class with
 constraint
    c_ok: $ forall nw/Employee s/Integer (nw salary s) ==> (s > 100) $
end
"
