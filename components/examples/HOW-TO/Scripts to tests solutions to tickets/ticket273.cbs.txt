# Check whether ConceptBase picks the right E-predicate
# Start: CBshell -f ticket273.cbs
#

startServer -u nonpersistent -port 4411 -t veryhigh
tell "CLASS in Class end
   Proposition with
      attribute
        inSubRel: CLASS
   end
   ISA in Class with
     single,necessary
      super: CLASS
     necessary
       sub: CLASS
  rule
      derive_inSubRel: $ forall x/Proposition spec/ISA SUBC/CLASS
                          (spec sub SUBC) and (x in SUBC) ==> (x inSubRel SUBC) $
   end

"
# CB should pick (spec sub SUBC) as E-predicate rather than (SUBC in CLASS)
# See line 'Cheapest binding path ...' in CBserver log
