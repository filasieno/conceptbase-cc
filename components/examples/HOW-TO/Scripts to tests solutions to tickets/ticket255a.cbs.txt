# Test whether an ECArule can be activated and de-acticated by another ECArule on the fly
# 
# This script requires ConceptBase 7.2 or later.
#

startServer -u nonpersistent -port 4411 -t veryhigh
tell "

Thread with
  attribute
    phase: Phase
end

Phase end

P0 in Phase end
P1 in Phase end
P2 in Phase end

mythread in Thread with
  phase
    p0: P0
end


Account with
  attribute
    balance: Integer
end

Transfer with
  attribute
    fromaccount: Account;
    toaccount: Account;
    amount: Integer
end 

{* NOTE, transferRule active FALSE *}

ECArule transferRule with
  mode m: Deferred
  active a: FALSE 
  rejectMsg rm:
   \"Account balance insufficient\"
  ecarule
        er : $  acc1,acc2/Account tr/Transfer m,b1,b2,nb1,nb2/Integer
        ON Tell (tr amount m) 
        IF `(tr fromaccount acc1) and
           `(tr toaccount acc2) and
           `(acc1 balance b1) and
           `(acc2 balance b2) and
           (nb1 = b1 - m) and
           (nb2 = b2 + m) and
           (nb1 >= 0)
        DO Retell (acc1 balance nb1) ,
           Retell (acc2 balance nb2) 
        ELSE
           reject
        $
end


{* the 2nd rule deduces a tax of 10 percent from both accounts participating in the transfer *}

ECArule taxRule with
  mode m: Deferred
  priority_after r1: transferRule
  rejectMsg rm:
   \"Account balance insufficient\"
  ecarule
        er : $  acc1,acc2/Account tr/Transfer m,b1,b2,nb1,nb2/Integer
        ON Tell (tr amount m) 
        IF `(tr fromaccount acc1) and
           `(tr toaccount acc2) and
           `(acc1 balance b1) and
           `(acc2 balance b2) and
            (b1 >= 0) and (b2 >= 0) and
           (nb1 = b1 - IDIV(b1,10) ) and
           (nb2 = b2 - IDIV(b2,10) ) 
        DO Retell (acc1 balance nb1) ,
           Retell (acc2 balance nb2) 
        ELSE
           reject
        $
end

ECArule activator with
  mode m: Immediate
  ecarule
        er : $  
        ON Tell (mythread phase P1) 
        IF `(transferRule active FALSE) 
        DO Retell (transferRule active TRUE) 
        $
end

ECArule deactivator with
  mode m: Immediate
  ecarule
        er : $  
        ON Tell (mythread phase P2) 
        IF TRUE
        DO Retell (transferRule active FALSE) 
        $
end
"


tell "
PetersAccount in Account with
  balance b: 500
end 

MarysAccount in Account with
  balance b: 100
end
"

tell "
mythread with
 phase
   p1: P1  {* this will trigger 'activator' to activate transferRule *}
end

transfer1 in Transfer with
   fromaccount a1: PetersAccount
   toaccount a2: MarysAccount
   amount m: 250
end
"
# both transferRule and taxRule should be executed via transfer1

tell "
mythread with
 phase
   p2: P2   {* this will trigger 'deactivator' to deactivate transferRule *}
end

transfer2 in Transfer with
   fromaccount a1: PetersAccount
   toaccount a2: MarysAccount
   amount m: 100
end
"
# only taxRule should be executed via transfer2

ask "get_object[PetersAccount/objname]" OBJNAMES  FRAME  Now
# should be 203
ask "get_object[MarysAccount/objname]" OBJNAMES  FRAME  Now
# should be 284
