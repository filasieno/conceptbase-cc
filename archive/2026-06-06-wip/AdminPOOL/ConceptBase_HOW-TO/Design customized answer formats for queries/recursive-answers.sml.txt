Class Person with
  attribute
    hasChild: Person
end



Person Adam with
  hasChild
    c1: Kain;
    c2: Abel;
    c3: Seth
end


Person Abel with
  hasChild
    c1: Abraham;
    c2: Lea
end

Person Abraham with
  hasChild
    c1: Isaak;
    c2: Ismael
end

Person Seth with
  hasChild
    c1: Lot
end

Person Kain with
end

Person Lea with
end

Person Isaak with
end

Person Ismael with
end

Person Lot with
end



View PersonQ isA Person with
  parameter
    p: Person
  inherited_attribute
    hasChild: Person
  constraint
    c: $ UNIFIES(~p,~this) $
end


AnswerFormat PersonLayout with
  forQuery fq: PersonQ
  order o: ascending
  orderBy ob: "this"

  pattern pt:
    "
    <person>{this}
      <children>
      {Foreach(({this.hasChild}),(s),{ASKquery(PersonQ[{s}/p],PersonLayout)})}
      </children>
    </person>
     "
end
