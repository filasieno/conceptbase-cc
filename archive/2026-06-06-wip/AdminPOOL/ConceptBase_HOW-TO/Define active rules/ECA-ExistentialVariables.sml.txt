{
* File: ECA-Existentialvariables.sml
* Author: Manfred Jeusfeld
* Created: 2010-06-07/M.Jeusfeld (2010-06-07/M.Jeusfeld)
* ------------------------------------------------------
* This example shows the use of query classes to deal with
* existentially quantified variables in ECArules,
*
}

Class Person with
  attribute
    name: String;
    marriedTo: Person
  rule
    symm: $ forall p1,p2/Person (p1 marriedTo p2) ==> (p2 marriedTo p1) $
end

Class Employee isA Person with
  attribute
    taxrate : Integer
end


{* We want to express in an ECArule that employees that are marriedTo a
   Person that is also an employee get taxrate 40 while employees married to
   a person that is no employee get taxrate 30. Employees that are not married
   get tax rate 50.
   The difficulty is that the IF-part of ECA rules does not support sub-expressions
   with existentially quantified variables, e.g.
      (exists e1/Employee (e marriedTo e1)
   The solution is to design query classes that do the quantification internally.
   This also yields more readable specifications.
*}

UnmarriedEmployee in QueryClass isA Employee with
  constraint
    c1: $ not exists p/Person (this marriedTo p) $
end

MarriedToEmployee in QueryClass isA Employee with
  constraint
    c1: $ exists e/Employee (this marriedTo e) $
end

MarriedToNonEmployee in QueryClass isA Employee with
  constraint
    c1: $ exists p/Person (this marriedTo p) and not (p in Employee) $
end




ECArule setTaxRate_1 with
mode m: Immediate
ecarule
        er : $ e/Employee
        ON Tell (e in Employee)
        IF `(e in UnmarriedEmployee)
        DO  Tell (e taxrate 50)
        $
end

ECArule setTaxRate_2 with
mode m: Immediate
ecarule
        er : $ e/Employee
        ON Tell (e in Employee)
        IF `(e in MarriedToEmployee)
        DO  Tell (e taxrate 40)
        $
end

ECArule setTaxRate_3 with
mode m: Immediate
ecarule
        er : $ e/Employee
        ON Tell (e in Employee)
        IF `(e in MarriedToNonEmployee)
        DO  Tell (e taxrate 30)
        $
end




{* example data

bill in Employee end  

mary in Employee with  
  marriedTo
    spouse: john
end

john in Person end

carl in Employee with 
  marriedTo 
    spouse: anne
end

anne in Employee end

*}
