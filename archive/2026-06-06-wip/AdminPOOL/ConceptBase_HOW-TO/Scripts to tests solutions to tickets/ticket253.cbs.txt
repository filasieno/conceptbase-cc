startServer -u nonpersistent -port 4411 -t veryhigh


# This example is (c) by Dominik Schmitz, Aachen!

tell "

{*
   Excerpt of i* framework
*}
MetaClass IStarObject with
  attribute
    delete : Boolean
end

MetaClass Element isA IStarObject    
end 

MetaClass IntentionalElement isA Element with
end

MetaClass TaskElement isA IntentionalElement with
end

MetaClass Link isA IStarObject with    
   attribute
      from : Element;
      to : Element
end

MetaClass DependencyLink isA Link with
end

MetaClass DecompositionLink isA Link with
   attribute
      from : TaskElement;
      to : IntentionalElement
end

{*
   Example setting
*}
SimpleClass Task1 in TaskElement with
end

SimpleClass Task2 in TaskElement with
end

SimpleClass DecompLink12 in DecompositionLink with
  from
    f : Task1
  to
    t : Task2
end

SimpleClass TaskDependum in TaskElement with
end

SimpleClass DepTaskR3TaskR3b in DependencyLink with
    from
       f : TaskDependum
   to
     t :  Task2
end

{*
   QueryClasses to be used in IF part of ECA rule later on
*}
QueryClass HasIncomingDependency isA Element with
constraint
  rule : $ exists dl/DependencyLink (dl to this) and not (dl delete TRUE)$
end

QueryClass HasIncomingDependencyTransitive isA Element with  
  constraint
    inDepRule : $ 
      (this in HasIncomingDependency) and not (this delete TRUE) or 
      (exists decL/DecompositionLink dest/Element 
              (decL from this) and (decL to dest) and not (dest delete TRUE) and (dest in HasIncomingDependencyTransitive))$
end 

QueryClass DelSuppNoIncDeps isA Element with
constraint
  rule : $ not (this in HasIncomingDependencyTransitive) $
end

QueryClass Deleted isA IStarObject with
constraint
  rule : $ (this delete TRUE) $
end

QueryClass NotDeleted isA IStarObject with
constraint
  rule : $ not (this delete TRUE) $
end

{*
   ECA rules
*}

DeleteElement_Links in ECArule with
{*  mode 
    m : Deferred *}
  ecarule
    er : $ x/Element l/Link
	ON Tell(A(x,delete,TRUE))
	IF (x in Element) and (l in Link) and `(l in NotDeleted) and ((l from x) or (l to x))
	DO Tell(A(l,delete,TRUE)) $
end 

DeleteDependencyLink in ECArule with  
{*  mode 
    m : Deferred*}
  ecarule
    er : $ x/DependencyLink y/Element
	ON Tell(A(x,delete,TRUE))
	IF (x in DependencyLink) and (x to y) and (y in IntentionalElement) and 
           `(y in DelSuppNoIncDeps) and {* Problem liegt here!!! *}
           `(y in NotDeleted)
	DO Tell(A(y,delete,TRUE)) $
end 


"


ask "DelSuppNoIncDeps" OBJNAMES  FRAME  Now


# This TELL forced ConceptBase into an infinite loop before the 
# correction  of ticket 253
tell "
TaskDependum with
  delete
    d : TRUE
end 
"


tell "
{* another example setting  *}
SimpleClass task1 in TaskElement with
end

SimpleClass task2 in TaskElement with
end

SimpleClass decompLink12 in DecompositionLink with
  from
    f : task1
  to
    t : task2
end

SimpleClass taskDependum in TaskElement with
end

SimpleClass depTaskR3TaskR3b in DependencyLink with
    from
       f : taskDependum
   to
     t :  task2
end

" 


tell "
{* this rule is not capable to delete the objects marked with 'delete' 
   since attributes of these objects are not deleted; we test it here because
   it shows whether the ECAevaluator can deal with evaluating predicates
   (x in c), where c is a free variable
*}
CleanUp in ECArule with  
  mode 
    m : Deferred
  ecarule
    er : $ x/IStarObject c/Proposition
	ON Tell(A(x,delete,TRUE))
	IF (x in c)
	DO Untell (x in c) $
end
"

# initiate test for CleanUp; it shall produce integrity violations but should not
# crash or enter an infinite loop
tell "
taskDependum with
  delete
    d : TRUE
end 
"
