startServer -u nonpersistent -port 4411 -t high


tell "

A end

C isA D end

D with
end

X1 in A,C end

Aclass with   {* needed since Untell (x attribute y) isn't fully supported *}
  attribute
    link: Aclass
end

aaa in Aclass with
  link
    a1: bbb
end

bbb in Aclass end

"



tell "
ECArule AtoC with
  mode m: ImmediateDeferred
  ecarule
        er : $  x/Proposition
        ON Untell (x in A) 
        IF TRUE
        DO Untell (x in C), Untell (aaa link bbb) 
        $
end


ECArule forD with
  mode m: ImmediateDeferred
  ecarule
        er : $  x/Proposition
        ON Untell (x in D) 
        IF TRUE
        DO Tell (D attribute x)
        $
end

ECArule forBBB with
  mode m: ImmediateDeferred
  ecarule
        er : $  x/Aclass
        ON Untell (x link bbb) 
        IF TRUE
        DO Tell (bbb link x)
        $
end

"


untell "
X1 in A end
"




ask "get_object[D/objname]" OBJNAMES  FRAME  Now
# should have X1 as attribute value 

ask "get_object[bbb/objname]" OBJNAMES  FRAME  Now
# should have aaa as link value
