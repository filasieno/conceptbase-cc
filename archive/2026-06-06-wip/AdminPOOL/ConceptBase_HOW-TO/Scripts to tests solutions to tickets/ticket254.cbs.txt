startServer -u nonpersistent -port 4411 -t high


tell "

A end

C isA D end

D with
end

X1 in A,C end


aaa  with
  attribute
    a1: bbb
end

bbb end

"



tell "
ECArule AtoC with
  mode m: ImmediateDeferred
  ecarule
        er : $  x/Proposition
        ON Untell (x in A) 
        IF TRUE
        DO Untell (x in C), Untell (aaa attribute bbb) 
        $
end


ECArule forD with
  mode m: ImmediateDeferred
  ecarule
        er : $  x/Proposition
        ON Untell (x in D) 
        IF TRUE
        DO Tell (D attribute x)
        $
end

ECArule forBBB with
  mode m: ImmediateDeferred
  ecarule
        er : $  x/Proposition
        ON Untell (x attribute bbb) 
        IF TRUE
        DO Tell (bbb attribute x)
        $
end

"


untell "
X1 in A end
"




ask "get_object[D/objname]" OBJNAMES  FRAME  Now
# should have X1 as attribute value 

ask "get_object[bbb/objname]" OBJNAMES  FRAME  Now
# should have aaa as link value
