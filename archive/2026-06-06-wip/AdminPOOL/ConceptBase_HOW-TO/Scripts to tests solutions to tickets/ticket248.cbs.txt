startServer -u nonpersistent -port 4411 -t high
tell "

Account with
  attribute
    balance: Integer
end

Transfer in Class with
  attribute
    fromaccount: Account;
    toaccount: Account;
    amount: Integer;
    participatingaccount: Account   {* either from- or toaccount *}
  rule
    r1: $ forall t/Transfer acc/Account 
            ((t fromaccount acc) or (t toaccount acc)) ==> (t participatingaccount acc) $
end 


{* The interesting predicate here is `(tr participatingaccount acc) 
   ConceptBase should be able to find all solutions of this
   predicate when the IF-part of the rule is evaluated.
   The mode of the ECArule has here no influence on the execution semantics.
*}

ECArule taxRule with
  mode m: ImmediateDeferred
  ecarule
        er : $  acc/Account tr/Transfer m,b,nb/Integer
        ON Tell (tr amount m) 
        IF `(tr participatingaccount acc) and
           `(acc balance b) and
           (nb = b - IDIV(b,10) )
        DO Retell (acc balance nb) 
        $
end
"


tell "
PetersAccount in Account with
  balance b: 500
end 

MarysAccount in Account with
  balance b: 100
end
"



tell "

{* This update triggers the ECArule and should make two solutions for
    both (transfer1 participatingaccount PetersAccount) and
    (transfer1 participatingaccount MarysAccount) true via the
   deductive rule r1
*}

transfer1 in Transfer with
   fromaccount a1: PetersAccount
   toaccount a2: MarysAccount
   amount m: 250
end
"



ask "get_object[PetersAccount/objname]" OBJNAMES  FRAME  Now
ask "get_object[MarysAccount/objname]" OBJNAMES  FRAME  Now
