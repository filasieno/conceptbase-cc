# simple test: PersonA should be recognized as the type of x,y,z and the expression "PersonA(A(x," should
# not be tested as a candidate type looking like a function call
# There should be no errors messages like 
#     Unable to find an object x
#     Unable to find an object hasAncestor

tell "
 PersonA in Class with  
  attribute
    hasAncestor : PersonA
  rule
    hasAncestor_transitive : $ forall x,y,z/PersonA   (A(x,hasAncestor,y) and (y hasAncestor z)) ==> (x hasAncestor z) $
end 

"

# this shows that type expressions like Interval[1,1000] also make some sense and should work
# There are certainly more elegant ways to map integers to levels, e.g. by a function
# The example here has the sole purpose to check that an expression like s/Interval[1,1000]
# is accepted and processed correctly
tell "
GenericQueryClass Interval isA Integer with
  parameter
     a: Integer;   
     b: Integer
  constraint
    c1: $ (a <= this) and (this <= b) $
end

Employee in Class with  
  attribute
    salary : Integer;
    level: Integer
  rule
    lev1rule : $ forall p/Employee s/Interval[1,1000] (p salary s) ==> (p level 1) $;
    lev2rule : $ forall p/Employee s/Interval[1001,5000] (p salary s) ==> (p level 2) $;
    levrestrule : $ forall p/Employee s/Integer (p salary s) and (s > 5000) ==> (p level 3) $
end 

LeveledEmployee in QueryClass isA Employee with
  retrieved_attribute
    level: Integer
end
"


tell "
bill in Employee with
  salary s: 500
end

mary in Employee with
  salary s: 1500
end

anne in Employee with
  salary s: 5500
end
"

#this query should return bill with level 1, mary with level 2 and anne with level 3.
ask "LeveledEmployee" OBJNAMES  FRAME  Now
