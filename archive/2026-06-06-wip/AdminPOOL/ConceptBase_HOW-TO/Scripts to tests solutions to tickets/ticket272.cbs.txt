startServer -u nonpersistent -port 4411 -t veryhigh
tell "
Proposition with
     attribute
       inViaMC: METACLASS
end

METACLASS end
"

tell "
METACLASS in Class with
      constraint
        exists_in_in: $ exists x/VAR C/METACLASS (x inViaMC C) $
end
"

# Another variant that also is problematic
# The label MC is undefined, so ConceptBase need to find this error
tell "
   Proposition with
     attribute
       inViaMC1: METACLASS1
   end

   METACLASS1 in Class with
      constraint
        exists_in_in: $ exists x/VAR C/MC (x inViaMC1 C) $
       rule
        aux_rule: $ forall x/VAR C/MC (x in C) ==> (x inViaMC1 C) $
   end
"
