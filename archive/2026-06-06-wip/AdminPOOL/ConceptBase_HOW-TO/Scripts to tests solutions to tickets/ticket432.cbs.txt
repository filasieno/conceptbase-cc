# Function calls not allowed in predicates
#

# cbserver -t low
connect

tell "
StringNext in Class with
  attribute
     string: String;
     nextstring: String
end
"

tell '
StringNext in Class with
  rule
     onnext: $ forall e/StringNext i/String (e string i) ==> (e nextstring ConcatenateStrings("s",i)) $
end
'
why

tell "
StringNext2 in Class with
  attribute
     string: String;
     nextstring: String
end
"

tell '
StringNext2 in Class with
  rule
     onnext2: $ forall e/StringNext2 i,s/String (e string i) and (s = ConcatenateStrings("s",i)) ==> (e nextstring s) $
end
'
why


tell '
n100 in StringNext with
  string n: "100"
end
n101 in StringNext2 with
  string n: "101"
end
'





tell "
NextStringOf2 in GenericQueryClass isA String with
  parameter,computed_attribute
    object: StringNext2
  constraint
    c1: $ (~object nextstring ~this) $
end
"

ask "NextStringOf2" OBJNAMES FRAME Now
showAnswer


tell "
NextStringQ2 in QueryClass isA StringNext2 with
  retrieved_attribute
     string: String;
     nextstring: String
end
"

ask "NextStringQ2" OBJNAMES FRAME Now
showAnswer
