# Check capabilities of ECArules to prescribe a certain order of actions
# Inspired by figure 4-10 of master thesis of L. Liu
#

startServer -u nonpersistent -port 4411 -t veryhigh



tell "

y1 end
y2 end

A in Class end
B in Class end
C in Class end
D in Class end
E in Class end

D1 in Class end
D2 in Class end



ECArule onA with
  mode m: Deferred
  ecarule
        er : $  x/Proposition
        ON Tell (x in A) 
        IF TRUE
        DO tBegin,
           Tell (x in B),
           Tell (x in C),
           tEnd
        $
end

ECArule onB with
  mode m: Deferred
  ecarule
        er : $  x/Proposition
        ON Tell (x in B) 
        IF TRUE
        DO Tell (x in D)
        $
end

ECArule onC with
  mode m: Deferred
  ecarule
        er : $  x/Proposition
        ON Tell (x in C) 
        IF TRUE
        DO Tell (x in E)
        $
end

ECArule onD with
  mode m: Immediate
  ecarule
        er : $  x/Proposition
        ON Tell (x in D) 
        IF TRUE
        DO Tell (x in D1),
           Tell (x in D2)
        $
end


ECArule onY with
  mode m: Immediate
  ecarule
        er : $ 
        ON Ask find_instances[Class/class] 
        IF TRUE
        DO Tell (y1 in A),
           Tell (y2 in A)
        $
end



"


tell "
x1 in A end
x2 in A end
"
