# Test ECArules for cascading UNTELL
# This version uses Immediate mode; you can also test
# the rules with other modes. They will lead to slightly
# different orderings of the cascading UNTELLs.
#
# Start: CBshell -f ticket296.cbs
#



startServer -u nonpersistent -t veryhigh -eo on

tell "
ECArule cascadeUntellInClass with
  mode m: Immediate
  ecarule
    r1: $ x/Individual c/Proposition
         ON Untell (x in Individual)
         IF In_s(x,c)
         DO Untell (x in c)
        $
end

ECArule cascadeUntellClassOf with
  mode m: Immediate
  ecarule
    r1: $ x/Individual c/Proposition
         ON Untell (c in Individual)
         IF In_s(x,c)
         DO Untell (x in c)
        $
end

ECArule cascadeUntellInattributeeCategory with
  mode m: Immediate
  ecarule
    r1: $ a/attributee ac/attributee
         ON Untell (a in attributee)
         IF In_s(a,ac)
         DO Untell (a in ac)
        $
end

ECArule cascadeUntellattributeeCategoryOf with
  mode m: Immediate
  ecarule
    r1: $ a/attributee ac/attributee
         ON Untell (ac in attributee)
         IF In_s(a,ac) 
         DO Untell (a in ac)
        $
end

ECArule cascadeUntellToAttr with
  mode m: Immediate
  ecarule
    r1: $ x/Individual a/attributee
         ON Untell (x in Individual)
         IF To(a,x) and (a in attributee)
         DO Untell (a in attributee)
        $
end

ECArule cascadeUntellFromAttr with
  mode m: Immediate
  ecarule
    r1: $ x/Individual a/attributee
         ON Untell (x in Individual)
         IF From(a,x) and (a in attributee)
         DO Untell (a in attributee)
        $
end

ECArule cascadeUntellHasSuper with
  mode m: Immediate
  ecarule
    r1: $ c/Individual sc/Individual
         ON Untell (c in Individual)
         IF Isa_e(c,sc)
         DO Untell (c isA sc)
        $
end

ECArule cascadeUntellHasSub with
  mode m: Immediate
  ecarule
    r1: $ c/Individual sc/Individual
         ON Untell (sc in Individual)
         IF Isa_e(c,sc)
         DO Untell (c isA sc)
        $
end

ECArule cascadeUntellHasSuperAC with
  mode m: Immediate
  ecarule
    r1: $ c/attributee sc/attributee
         ON Untell (c in attributee)
         IF Isa_e(c,sc)
         DO Untell (c isA sc)
        $
end

ECArule cascadeUntellHasSubAC with
  mode m: Immediate
  ecarule
    r1: $ c/attributee sc/attributee
         ON Untell (sc in attributee)
         IF Isa_e(c,sc)
         DO Untell (c isA sc)
        $
end
"


# Sample sample classes
tell "
Employee in Class with
  attribute
     name: String;
     salary: Integer
end

HighSalary isA Integer end

Manager in Class isA Employee with
  attribute
    salary: HighSalary
end
"

# sample instance of Manager
tell "
1000 in HighSalary end

bill in Manager with
  name
     bname: \"William\"
  salary
     bsal: 1000
end "

ask "get_object[bill/objname]" OBJNAMES  FRAME  Now

# if we untell just bill, the ECArules will untell its links as well
untell "
bill in Individual 
end
"

# so, bill should be unknown after the untelling
ask "get_object[bill/objname]" OBJNAMES  FRAME  Now


# an instance of Employee
tell "
mary in Employee with
  name
     mname: \"Marianne\"
  salary
     nsal: 200
end "

# now we untell the class of mary
untell "
Employee in Individual 
end
"

# it will not only delete Employee but also its subclass link to Manager,
# its attributes, and all links from mary to Employee
# Afterwards, mary is a "naked" object without any reference to Employee

ask "get_object[Employee/objname]" OBJNAMES  FRAME  Now
ask "get_object[mary/objname]" OBJNAMES  FRAME  Now
