startServer -u nonpersistent -port 4411 -t veryhigh

tell "

y1 end
y2 end

A in Class end
B in Class end
C in Class end
D in Class end
E in Class end

D1 in Class end
D2 in Class end


ECArule onA with
  mode m: Deferred
  ecarule
        er : $  x/Proposition
        ON TRANSACTIONAL Tell (x in A) 
        DO Tell (x in B),
           Tell (x in C)
        $
end


ECArule onB with
  mode m: Deferred
  ecarule
        er : $  x/Proposition
        ON Tell (x in B) 
        DO Tell (x in D)
        $
end

ECArule onC with
  mode m: Deferred
  ecarule
        er : $  x/Proposition
        ON Tell (x in C) 
        DO Tell (x in E)
        $
end

ECArule onD with
  mode m: Immediate
  ecarule
        er : $  x/Proposition
        ON Tell (x in D) 
        DO Tell (x in D1),
           Tell (x in D2)
        $
end

"


tell "
x1 in A end
x2 in A end
"
