# Check wehther ECA rules are correctly stored persistently

startServer -d /tmp/DBECA -port 4411 -t high

tell "
Number end

ECArule ecarule1 with
  mode m: Deferred
  ecarule
        er : $  x/Proposition
        ON Tell (x in Integer) 
        IF (x > 10)
        DO Tell (x in Number)
        $
end
"

stopServer

startServer -d /tmp/DBECA -port 4411 -t high

ask "get_object[ecarule1/objname]" OBJNAMES  FRAME  Now


tell "11 in Integer end"

ask "get_object[11/objname]" OBJNAMES  FRAME  Now
