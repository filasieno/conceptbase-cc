# Extended example showing that transitive+symmetric should better not be used in 
# combination for the same attribute; instead one should define an auxiliary attribute 
# and define it as the transitive closure of the symmetric attribute; 
# see definition of ModelElement and ModelElement2
# 
# We use cache mode 'transient' to make sure that all queries are evaluated against an
# initially empty cache. Otherwise, a subsequent query call could appear to be fast but
# in fact benefits from the cache filled by the preceding query call.
# The example is told in two steps. The first part is a model called INFOFLOW. It 
# features the meta meta class NodeOrLink, a meta class level (e.g. Person), and a class level
# (Dean,...). The second tell operation defines a class ModelElement (resp. ModelElement2)
# that includes an attribute linkto. This attribute is the one to be declared
# symmetric+transitive. 
#
# You can run the test script by:  CBshell -f ticket203.cbs

{* Some generic constructs applicable to any context *}
{* M. Jeusfeld, 4-Oct-2007 *}

{* single: declare an attribute to be single-valued (at most one value)
   necessary: declare an attribute to be required (at least one filler)
   reflexive: any x stands in relation M to itself (e.g. 'knows')
   transitive, symmetric, asymmetric: see definitions on Wikipedia
*}


Proposition isA Proposition with  
attribute
    reflexive : Proposition;
    transitive : Proposition;
    symmetric : Proposition;
    asymmetric : Proposition;
    single : Proposition;
    necessary : Proposition
end 


RelationSemantics in Class with  
constraint
    asym_IC : $ forall AC/Proposition!asymmetric C/Proposition x,y/VAR M/VAR
                     P(AC,C,M,C) and (x in C) and (y in C) and
                     (x M y)  ==> not (y M x) $
rule
    trans_R : $ forall x,z,y,M/VAR 
                     AC/Proposition!transitive C/Proposition
                     P(AC,C,M,C) and (x in C) and (y in C) and (z in C) and
                     (x M y) and (y M z) ==> (x M z) $;
    refl_R : $ forall x,M/VAR 
                    AC/Proposition!reflexive C/Proposition
                    P(AC,C,M,C) and (x in C)
                      ==> (x M x) $;
    symm_R : $ forall x,y,M/VAR 
                    AC/Proposition!symmetric C/Proposition
                    P(AC,C,M,C) and (x in C) and (y in C) and
                    (x M y)  ==> (y M x) $
end 

AdditionalConstraints in Class with  
constraint
    singleConstraint : $ forall c,d/Proposition p/Proposition!single x,m/VAR
              P(p,c,m,d) and (x in c) ==>
                (
                  forall a1,a2/VAR
                    (a1 in p) and (a2 in p) and Ai(x,m,a1) and Ai(x,m,a2) ==>
                   (a1=a2)
                ) $;
    necConstraint : $ forall c,d/Proposition p/Proposition!necessary x,m/VAR
            P(p,c,m,d) and (x in c) ==>
             exists y/VAR (y in d) and (x m y) $
end

tell "
{ 
*
* File: ModElem.sml
* Author: Manfred Jeusfeld
* Creation: 13-Nov-2008 (13-Nov-2008/MJf)
* ----------------------------------------------------------------------
* This file allows to perfrom connectivity analysis for models at the M1:ML level
* (models about systems). It is important not to define the attribute linkto as
* both symmetric and transitice. Instead, we only make linkto symmetric and then define
* linkto_trans as the transitive closure of linkto. Doing this in one step would
* be formally correct but creates very inefficient code.
}


{* placeholder for NodeOrLink defined for the M3:NDL level *}
NodeOrLink with
  attribute
    connectedTo: NodeOrLink
end 

ModelElement in Class with
  symmetric
    linkto: ModelElement
  attribute
    linkto_trans: ModelElement
  rule
    me1: $ forall x/VAR (x [in] NodeOrLink)  ==> (x in ModelElement) $;
    me2: $ forall link/VAR (link [in] NodeOrLink!connectedTo) ==> (link in ModelElement!linkto) $;
    me3: $ forall x/ModelElement link/ModelElement (link in ModelElement!linkto) and From(link,x) ==> (x linkto link) $;
    copy: $ forall x,y/ModelElement (x linkto y) ==> (x linkto_trans y) $;
    trans: $ forall x,y,z/ModelElement (x linkto z) and (z linkto_trans y) ==> (x linkto_trans y) $
end


{* inefficient version (uses symmetric+transitive in combination)  *}
ModelElement2 in Class with
  symmetric,transitive
    linkto: ModelElement2
  rule
    me1: $ forall x/VAR (x [in] NodeOrLink)  ==> (x in ModelElement2) $;
    me2: $ forall link/VAR (link [in] NodeOrLink!connectedTo) ==> (link in ModelElement2!linkto) $;
    me3: $ forall x/ModelElement2 link/ModelElement2 (link in ModelElement2!linkto) and From(link,x) ==> (x linkto link) $
end

UnconnectedModelElement in GenericQueryClass isA ModelElement with  
  parameter,computed_attribute
    start : ModelElement
  constraint
    uc : $ not (start linkto_trans this) $
end 

ConnectedModelElement in GenericQueryClass isA ModelElement with
  parameter,computed_attribute
    start : ModelElement
  constraint
    uc : $ (start linkto_trans this) $
end 

{* Test queries for the inefficient version *}
UnconnectedModelElement2 in GenericQueryClass isA ModelElement2 with
  parameter,computed_attribute
    start : ModelElement2
  constraint
    uc : $ not (start linkto this) $
end

ConnectedModelElement2 in GenericQueryClass isA ModelElement2 with
  parameter,computed_attribute
    start : ModelElement2
  constraint
    uc : $ (start linkto this) $
end 





"
