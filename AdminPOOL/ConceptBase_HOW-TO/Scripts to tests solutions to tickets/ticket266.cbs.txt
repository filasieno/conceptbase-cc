# Tests whether the EQ (=) and NE (<>) predicates are correctly processed in meta formulas
# Start: CBshell -f ticket264.cbs
#

startServer -u nonpersistent -port 4411 -t high
tell "

EntityType with
end 


ISA_Type  with  
  attribute
    super : EntityType;
    sub : EntityType
end 


ISA_Type_disjoint isA ISA_Type  
end 


"

tell "
ISA_Type_disjoint in Class with  
  constraint
    disjC : $ forall x/VAR ISAD/ISA_Type_disjoint e1,e2/EntityType (ISAD sub e1) and (ISAD sub e2) and (e1 <> e2) and (x in e1)
                     ==> not (x in e2) $
end 
"

tell "
Organization in EntityType end
PublicOrganization in EntityType end
PrivateOrganization in EntityType end
ISA3 in ISA_Type_disjoint with
  super
    sup1 : Organization
  sub
    sub1 : PublicOrganization;
    sub2 : PrivateOrganization
end 
"

#This should lead to an integrity violation:
tell " 
UvT in PublicOrganization,PrivateOrganization
end 
" 

ask "get_object[PrivateOrganization/objname]" OBJNAMES  FRAME  Now
ask "get_object[PublicOrganization/objname]" OBJNAMES  FRAME  Now
