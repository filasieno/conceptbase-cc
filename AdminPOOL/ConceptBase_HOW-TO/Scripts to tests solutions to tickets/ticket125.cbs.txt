NodeOrLink with
  attribute
     connectedTo: NodeOrLink

NodeOrLink!connectedTo isA NodeOrLink end

Model isA Node with
  attribute

ObjectType in Node end  
EntityType in Node isA ObjectType end

RelationshipType in Node isA ObjectType with
  connectedTo
     role: EntityType

ERDModel in Model with
  contains
    containsType: ObjectType
end

Set in Node, Class with
  rule
    r1: $ forall o1/OrgUnit o2/OrgUnit ((o2 haspredecessor o1) and (o1 in ~this)) ==>
		  (o2 in ~this) $

OrgUnit in Node, Class with
  rule
    r1: $ forall o/OrgUnit (o haspredecessor o) $;
	r2: $ forall o1/OrgUnit o2/OrgUnit (o1 generallink o2) ==> (o2 haspredecessor o1) $;
	r3: $ forall o1/OrgUnit o2/OrgUnit o3/OrgUnit ((o3 haspredecessor o2) and (o2 haspredecessor o1)) ==> 
		  (o3 haspredecessor o1) $
		
  connectedTo
    haspredecessor: OrgUnit;
	generallink: OrgUnit;
    valueadding: OrgUnit;
    infosupply: OrgUnit;
    supervisory: OrgUnit

{* CHAIN *}

FromChainUnit in GenericQueryClass isA OrgUnit with
  parameter
    unit: OrgUnit
  constraint
    c1: $ (~this valueadding ~unit) $
end

ToChainUnit in GenericQueryClass isA OrgUnit with
  parameter
    unit: OrgUnit
  constraint
    c1: $ (~unit valueadding ~this) $
end

ChainUnit in QueryClass isA OrgUnit with
  constraint
   ChainConstraint: $ (COUNT[FromChainUnit[~this/unit]/class] = 1) and (COUNT[ToChainUnit[~this/unit]/class] = 1) $
end

{* HUB *}

FromHubUnit in GenericQueryClass isA OrgUnit with
  parameter
    unit: OrgUnit
  constraint
    c1: $ (~this generallink ~unit) $
end

ToHubUnit in GenericQueryClass isA OrgUnit with
  parameter
    unit: OrgUnit
  constraint
    c1: $ (~unit generallink ~this) $
end

HubUnit in QueryClass isA OrgUnit with
  constraint
    HubConstraint: $ ( ((COUNT[FromHubUnit[~this/unit]/class] >= 2) and 
						(COUNT[ToHubUnit[~this/unit]/class] >= 1)  ) or 
					   ((COUNT[FromHubUnit[~this/unit]/class] >= 1) and 
						(COUNT[ToHubUnit[~this/unit]/class] >= 2)) ) $
end
