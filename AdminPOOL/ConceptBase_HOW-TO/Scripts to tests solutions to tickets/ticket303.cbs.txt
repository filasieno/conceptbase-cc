# our standard integrity constraint and rule
tell "
Employee with
   rule
      BossRule : $ forall e/Employee m/Manager
                   (exists d/Department
                     (e dept d) and (d head m))
                   ==> (e boss m) $
end

Employee with
   constraint
      SalaryBound :
          $ forall e/Employee b/Manager x,y/Integer 
              ((e boss b) and 
              (e salary x) and
              (b salary y))
               ==> (x  <=  y) $
end 
"


tell "
Individual Jango in Employee with
  dept
    JangosLabor : Forschung
  salary 
    JangosSalary : 500000
end
"
# should fail because it violates salaryBound via BossRule


untell "
Employee with
   attribute,constraint
     SalaryBound :
          $ forall e/Employee b/Manager x,y/Integer 
              ((e boss b) and 
              (e salary x) and
              (b salary y))
               ==> (x  <=  y) $
end

"


tell "
Individual Jango in Employee with
  dept
    JangosLabor : Forschung
  salary 
    JangosSalary : 500000
end
"
# should succeed because the salaryBound constraint has been deleted before


tell "
Employee with
   attribute,constraint
     SalaryBound :
          $ forall e/Employee b/Manager x,y/Integer 
              ((e boss b) and 
              (e salary x) and
              (b salary y))
               ==> (x  <=  y) $
end
"
# should fail because we now have Jango in the database 


untell "
Jango with
  attribute,salary 
    JangosSalary : 500000
end
"
# should succeed: removes the problematic salary


tell "
Employee with
   attribute,constraint
     SalaryBound :
          $ forall e/Employee b/Manager x,y/Integer 
              ((e boss b) and 
              (e salary x) and
              (b salary y))
               ==> (x  <=  y) $
end

Employee!SalaryBound with
  comment
    hint: \"The salary of this employee is higher than the salary of one of his bosses. This may not be. \"
end

# Another constraint that depends on the BossRule

# A new rule deriving instances of PoorEmployee
tell "
PoorEmployee in Class with
  rule
     rPoor: $ forall e/Employee s/Integer (e salary s) and (s < 80000) ==> (e in PoorEmployee) $
end
"

# A rule that depends on rPoor; deleting rPoor shall lead to more solutions via rRich
tell "
RichEmployee in Class with
  rule
     rRich: $ forall e/Employee not (e in PoorEmployee) ==> (e in RichEmployee) $
end
"

# this makes sure that PoorEmployee has some instance even when the rule rPoor is untold
tell "Eleonore in PoorEmployee end"


# a constraint that depends on rRich
tell "
Employee with
  constraint
     cRich: $ forall e/Employee s/Integer (e salary s) and (e in RichEmployee) ==> (s > 40000) $
end
"

ask "find_instances[RichEmployee/class]" OBJNAMES  LABEL  Now



untell "
PoorEmployee with
  rule,attribute
     rPoor: $ forall e/Employee s/Integer (e salary s) and (s < 80000) ==> (e in PoorEmployee) $
end
"
# this untell should fail because the deletion of rPoor leads to a violation of cRich via the rule rRich

ask "find_instances[RichEmployee/class]" OBJNAMES  LABEL  Now
