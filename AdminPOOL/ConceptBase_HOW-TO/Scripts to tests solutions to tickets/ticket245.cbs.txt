startServer -u nonpersistent -p 4411 -t high
tell "

Account with
  attribute
    balance: Integer
end

Transfer with
  attribute
    fromaccount: Account;
    toaccount: Account;
    amount: Integer
end
    

ECArule transferRule with
  mode m: Deferred
  rejectMsg rm:
   \"Account balance insufficient\"
  ecarule
        er : $  acc1,acc2/Account tr/Transfer m,b1,b2,nb1,nb2/Integer
        ON Tell (tr amount m) 
        IF `(tr fromaccount acc1) and
           `(tr toaccount acc2) and
           `(acc1 balance b1) and
           `(acc2 balance b2) and
           (nb1 = b1 - m) and
           (nb2 = b2 + m) and
           (nb1 > 0)
        DO Retell (acc1 balance nb1) ,
           Retell (acc2 balance nb2)
        ELSE
           reject
        $
end
"

tell "
PetersAccount in Account with
  balance b: 500
end

MarysAccount in Account with
  balance b: 100
end
"


# two transfers in one transction: should fail
tell "
transfer1 in Transfer with
   fromaccount a1: PetersAccount
   toaccount a2: MarysAccount
   amount m: 450
end

transfer2 in Transfer with
   fromaccount a1: PetersAccount
   toaccount a2: MarysAccount
   amount m: 150
end
"

# first transfer a s single transaction: should succeed
tell "
transfer1 in Transfer with
   fromaccount a1: PetersAccount
   toaccount a2: MarysAccount
   amount m: 450
end
"

# second transfer a s single transaction: should fail
tell "
transfer2 in Transfer with
   fromaccount a1: PetersAccount
   toaccount a2: MarysAccount
   amount m: 150
end
"

ask "get_object[PetersAccount/objname]" OBJNAMES  FRAME  Now
ask "get_object[MarysAccount/objname]" OBJNAMES  FRAME  Now
