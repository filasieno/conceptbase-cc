# This CBshell script tests the ability of ConceptBase to derive all solutions of (x m y)
# including those where instantiation to the attribute category cc is derived.
# The class cc is the concerned class of the attribution predicate. For example, the
# concerned class of $(x linkto y)$ would be Element!linkto provided that x is an instance of
# Element. 
# Rule r1 derives instantiations to Element!linkto. We then expect that 
# (Employee linkto Department) is true.
# See also ticket 202.

startServer -u nonpersistent
#enrollMe localhost 4001

tell "
Element with
  attribute
    linkto: Element
end

Employee in Class,Element with
  attribute
    worksFor: Department
end

Department in Class,Element end

MyRule in Class with
  rule
    r1: $ forall e/Element link/Attribute From(link,e) ==> (link in Element!linkto) $
end
"

tell "
LinkedElement in QueryClass isA Element with
  constraint
    c1: $ exists e/Element (e linkto this) $ 
end
"

# should return Department
ask "LinkedElement" OBJNAMES  FRAME  Now
