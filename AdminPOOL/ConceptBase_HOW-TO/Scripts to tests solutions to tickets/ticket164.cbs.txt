   Employee in Class with
     attribute
       dept: Department;
       deptsize: Integer
     rule
       sizerule: $ forall e/Employee d/Department s/Integer n/Label
                         (e dept d) and Label(d,n) and (d size s) ==> (e deptsize/n s) $
   end
   Department with
     attribute
       size: Integer
   end
   "
   tell "

   end

   "

   end
