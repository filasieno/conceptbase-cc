{
*
* File: 01-ecamail.sml
* Author: Manfred Jeusfeld
* Creation: 21-Aug-2007 (4-Sep-2007)
* ----------------------------------------------------------------------
*
* Example of how to trigger an external shell program like sendmail
* via an active rule. The scenario is taken from conference management:
* each time an abstract submission is entered into ConceptBase, an
* active rule is activated to pass the content of the abstract submission
* to those members that have subscribed to this type of information.
* See file README.txt for more details.
*
* Be sure to edit the instances of Member to be able to test the example.
* This example requires ConceptBase 7.1 released after 4-Sep-2007.
* 
*}

Class InfoType end

Class FormType isA InfoType with
  attribute
    mailSubject: String;   {* to be used for compiling the mail content *}
    mailIntro: String
end


{* HTML_Forms are just combinations of strings submitted to the system *}
Class HTML_Form with
  attribute
    feature: String
end


{* Members subscribe to a certain info type and will want to receive an email *}
{* for each HTML form that has the appropriate form type.                     *}
Class Member with
  attribute
    fullname: String;
    wantsObjectMail: HTML_Form;
    subscribedTo: InfoType
  rule
    omr: $ forall mem/Member h/HTML_Form
               (exists t/FormType
                 AL(h,feature,form_type,t) and
                 (mem subscribedTo t) )
               ==> (mem wantsObjectMail h) $
end


{* This query is used to compile the mail text, i.e. all features of  *}
{* an HTML form plus the headers for the mail.                        *}
produceMailText in GenericQueryClass isA HTML_Form with
  parameter
    whichform: HTML_Form;
    receiver: Member
  retrieved_attribute
    feature: String
  computed_attribute
    mailIntro: String;
    mailSubject: String
  constraint
    c1: $ exists t/FormType (~this = ~whichform) and (~this feature/form_type t) and
          (t mailIntro ~mailIntro) and (t mailSubject ~mailSubject) $
end

{* The answer format takes care for the pretty printing of the answer to *}
{* the produceMailText query.                                            *}
{* The first two line contain the mail receiver and the mail subject.    *}
{* The next line contains the form identifier. The subsequent lines are  *}
{* built for each feature attribute of the answer object (ie. instance   *}
{* of HTML_Form. The UQ function removes quotes from the attribute       *}
{* values. The string \\n stands for a line feed.                        *}
{* A pretty-printed example looks like                                   *}
{*                                                                       *}
{*    The following text has been inserted ...                           *}
{*                                                                       *}
{*    To: xyz@abc.anywhere.org                                           *}
{*    Subject: AIM-07                                                    *}
{*    ******************                                                 *}
{*    form identifier: A1                                                *}
{*    memberPC: yes                                                      *}
{*    content: This ...                                                  *}
{*    ******************                                                 *}
{*    time extracted: 2007/08/22 13:01:45 (GMT)                          *}
{*    .                                                                  *}
{* The last line contains only a single dot. This will indicate to       *}
{* sendmail that the mail text is complete.                              *}


AnswerFormat MailFormat with
   forQuery q: produceMailText
   order o: ascending
   orderBy ob: "this"
  pattern p:
"To: {UQ({receiver})}
Subject: {UQ({this.mailSubject})}

{UQ({this.mailIntro})}

******************
form identifier: {this}
{Foreach(({this.feature},{this|feature}),(p,r),{r}: {UQ({p})}\\n)}******************
time extracted: {transactiontime} (GMT)\\n.
"
end


{* This is the active rule that is triggered by every insertion of a HTML form *}
{* All subscribers shall get an email with a pretty-print of the contents of   *}
{* the HTML form.                                                              *}
ECArule MailObjectOnFormSubmission with
  mode m: Deferred
  ecarule
    r1: $ h/HTML_Form mem/Member 
        ON Tell((h in HTML_Form))
        IF `(mem wantsObjectMail h)
        DO CALL(sendObject(h,mem))
        ELSE commit $
end


{* Example data *}

FormType "AbstractSubmission" with
  mailSubject subject: "AIM'07/AbstractSubmission"
  mailIntro introtext:
"The following object has been inserted into the database."
end


{* Replace here the email address by some valid email address, e.g. your own *}
Member "johndoe@earth.universe.org" with
  fullname fn: "John Doe"
  subscribedTo c1: "AbstractSubmission"
end
