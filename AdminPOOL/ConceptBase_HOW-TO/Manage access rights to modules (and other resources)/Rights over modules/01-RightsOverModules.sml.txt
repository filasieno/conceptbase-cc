 * Example rights model for all to be told to module 'oHome'.
 * Be sure that you start the CBserver with option -s 2 to
 * enable the access control.
*}



Mjeusfeld in Module end
Mjonny in Module end

CB_User jeusfeld with
  homeModule h: Mjeusfeld
end

CB_User jonny with 
  homeModule h: Mjonny
end

{* We only allows read operations in the oHome module *} 
GenericQueryClass CB_PermittedoHome isA CB_User with
  parameter
    user: CB_User;
    res: Resource;
    op: CB_Operation
  constraint
    cperm: $
              (~op in CB_ReadOperation) and 
              UNIFIES(~user,~this) 
           $
end
