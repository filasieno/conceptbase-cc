# script for extracting filenames and their sizes from the current 
# directory and converting them into CBshell commands to store this info
# in ConceptBase
# Usage:
#   gendirsize > /tmp/script.cbs
#   cbshell /tmp/script.cbs
# or
#   gendirsize | cbshell -p
#
# Rather than using gendiresize directly, you can also use the 'dirsize' script.
#

# author of the source code.
# Any re-distribution as source code must acknowledge the original author of this file.

echo "startServer -port 4001 -t no" 

#tell ConceptBase the required classes for the dirSize function
echo "tell \"Directory in Class with
  attribute
    hasFileSize: Integer
end
currDir in Directory end

File in Class with
  attribute
    size: Integer
end

Directory in Class with
  rule
    fsRule: $ forall f/File fn/Label s/Integer  (f size s) and Label(f,fn)
                  ==> (currDir hasFileSize/fn s) $
end

dirSize in Function isA Integer with
  constraint
    c: $ (this = SUM_Attribute(Directory!hasFileSize,currDir)) $
end\"
" 

# sleep 1

echo tell \" 

for file in *
do
  if [ -f "${file}" ]; then  
     fsize=$( stat -c %s "${file}")
     echo \\\"${file}\\\" in File with size s: ${fsize} end
    fi
done

echo \" 

echo ask \"dirSize\" OBJNAMES  LABEL  Now
echo showAnswer
