
{ * This add a new graphical type "RedArrowGT" to the pre-defined }
{ * palette of graphical types. A second new graph type is added  }
{ * for a thick red line.                                         }
{ *                        .-- Manfred Jeusfeld, 9-Mar-1998       }

{ * This graphical palette lists the graphical types to be used   }
{ * when displaying instances of Hints and Arguments. The         }
{ * graphical types c1..c8 and d are like in the default palette  }
{ * X11GraphBrowserPalette. However, the rule for c8 is redefined }
{ * in order to cope with the special display of the attachedTo   }
{ * attribute used in the example below. Note that for each object}
{ * x there may only be one graphical type gt within a palette for}
{ * which (x graphtype gt) is true.                               }
{ * The graph types a1 and a2 are new ones. The first is for the  }
{ * icon of Hint. The second is for the attribute attachedTo.     }


GraphicalPalette HintPalette with 
  attribute,contains
     c1 : TokenGtype;
     c2 : MetametaGtype;
     c3 : SimpleGtype;
     c4 : MetaGtype;
     c5 : ClassGtype;
     c6 : InGtype;
     c7 : IsAGtype;
     c8 : AGtype_1;   
     a1 : RedArrowGT;
     a2 : ThickRedLineGT
  attribute,default
     d : PropositionGtype
end 


{ * This replaces the AGtype and defines the display of all attributes }
{ * except those which are instances of Hint!attachedTo.               }

Class AGtype_1 in ATK_Lineedge with 
  rule
     gtrule : $ forall e/Attribute (e in Attribute) and 
                  (not (e in Hint!attachedTo)) ==>
                  (e graphtype AGtype_1)$
  WithShape
     shape : lineedge
  TextColor
     tcolor : black
  LineColor
     lcolor : MediumSeaGreen
  LineWidth
     lwidth : 1
  FontDesc
     font : andysans8
  LineCap1
     lcap1 : 2
  LineCap2
     lcap2 : 4
end 

{ * This defines RedArrowGT with the icon redarrow.xpm.           }
{ * Note that the file redarrow.xpm must be in the directory      }
{ * ~/bitmaps or in the directory $CB_HOME/lib/bitmaps.           }
{ * Both paths must be readable for the CBinterface.              }
{ * The rule is used to define for which Telos objects x the      }
{ * the new graphical type should be used.                        }

Class RedArrowGT in ATK_Iconnode with
  rule
    gtrule: $ forall x/Proposition (x in Hint) ==> (x graphtype RedArrowGT) $
  WithShape
    shape : iconnode
  PixmapFile
    file  : "redarrow.xpm"
end

{ * This defines ThickRedLineGT. Note that we had to make sure    }
{ * that the graphical type AGtype_1 is only used for ordinary    }
{ * attributes but not for instances of Hint!attachedTo.          }
{ * In principle, we can have very complex rules for assigning    }
{ * graphical types to nodes and links. We have however to make   }
{ * sure that for any object x there is only one graphical type   }
{ * gt in the selected palette for which (x graphtype gt) holds.  }

Class ThickRedLineGT in ATK_Lineedge with
  rule
    gtrule: $ forall a/Attribute (a in Hint!attachedTo) ==>
                (a graphtype ThickRedLineGT) $
  WithShape
    shape   : lineedge
  TextColor
    tcolor  : red { <-- should be a legal color from X11 color map }
  LineColor
    lcolor  : red
  LineWidth
    lwidth  : 4  { <--- this is thje thickness }
  FontDesc
    font    : andysans10
  LineCap1
    lcap1   : 2
  LineCap2
    lcap2   : 4
end
