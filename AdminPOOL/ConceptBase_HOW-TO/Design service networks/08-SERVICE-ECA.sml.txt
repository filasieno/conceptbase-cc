AnyObject end

MappableObject isA AnyObject end

EconomicResourceType isA MappableObject with
end

EntityType isA AnyObject with
end

RelationshipType isA AnyObject with
end





Mapping with
 attribute
  input: Proposition;
  output: Proposition
end



UnMapped in QueryClass isA MappableObject with
  constraint
    c1: $ not exists m/Mapping o/AnyObject (m input ~this) and (m output o) $
end

MappedTo in QueryClass isA MappableObject with
  computed_attribute
    target: AnyObject
  constraint
    c1: $ exists m/Mapping o/AnyObject (m input ~this) and (m output ~target) $
end


Map_R1 isA Mapping with
  attribute
    servicetype: ServiceType;
    entitytype: EntityType
end


deriveStoredAs in Class with
  rule
   r1: $ forall er/EconomicResourceType E/EntityType m/Mapping
            (m input er) and (m output E) ==> (er storedAs E) $;
   r2: $ forall sf/EconomicResourceType!stockflow E/ObjectType m/Mapping
            (m input sf) and (m output E) ==> (sf storedAs E) $
end



Map_R1!servicetype isA Mapping!input end
Map_R1!entitytype isA Mapping!output end



ECArule Perform_Map_R1 with
  mode m: Deferred
  ecarule
        er : $  map/Map_R1 st/ServiceType S2/String
        ON Tell (map in Map_R1) 
        IF `(map servicetype st) and
           `(st in complement[R1_ServiceType_WithEntityType])  and
           (S2 = StringToLabel(ConcatenateStrings("E_",st))) 
        DO Tell (S2 in EntityType),
           Tell (map entitytype S2) 
        ELSE
           reject
        $
end



{*

MyExchange in ServiceType end
MyProduce in ServiceType end


map1 in Map_R1 with
  servicetype 
    e1: MyExchange;
    e2: MyProduce
end
