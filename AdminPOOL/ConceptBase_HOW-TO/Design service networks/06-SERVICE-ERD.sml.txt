ObjectType in Concept end



EntityType in Concept isA ObjectType with
  connectedTo
    entAttr: Domain
end

Domain in Concept end

NumberDomain in Concept isA Domain end

INT in NumberDomain end
FLOAT in NumberDomain end

RelationshipType in Concept isA ObjectType with
  connectedTo
     role: EntityType
end




RelationshipType!role with
  connectedTo,single
     card: Cardinality
end

Cardinality in Node end

{* we support here just three cardinalities *}
"1..1" in Cardinality end
"0..1" in Cardinality end
"1..*" in Cardinality end

{* this is a sgortcut for relationship types with 1..1 cardinality *}
RelationshipType_1_1 in Concept isA RelationshipType end
