ProgramModule in Class with
  attribute
    imports: ProgramModule;
    uses: ProgramModule
end

ECArule Imports2Uses with
  mode m: Deferred
  ecarule r: $ m1,m2/ProgramModule
             ON Tell (m1 imports m2)
             IF TRUE
             DO Tell (m1 uses m2) $
end



{*  -- and now the GraphViz translator *}



QueryClass ShowNodes isA ProgramModule end
QueryClass ShowImportLinks isA ProgramModule!imports end
QueryClass ShowUseLinks isA ProgramModule!uses end



GenericQueryClass imports2GV isA YesClass with
end
GenericQueryClass uses2GV isA YesClass with
end


ImportsInGraphViz in AnswerFormat with
  forQuery q: imports2GV
  head h: 
"# Generated by ConceptBase {cb_version} at {transactiontime}
# Process this file by Graphviz, e.g. 
#    dot -Tpng thisfile.txt > thisfile.png

"
  pattern p:
"digraph ProgramModule \{
{ASKquery(ShowNodes,ELLIPSENODE_FORMAT)}
{ASKquery(ShowImportLinks,LINK_FORMAT)}
overlap=false
label=\"Program module imports \\\nExtracted from ConceptBase and layed out by Graphviz \"
fontsize=12;
\}
"
end


UsesInGraphViz in AnswerFormat with
  forQuery q: uses2GV
  head h: 
"# Generated by ConceptBase {cb_version} at {transactiontime}
# Process this file by Graphviz, e.g. 
#    dot -Tpng thisfile.txt > thisfile.png

"
  pattern p:
"digraph ProgramModule \{
{ASKquery(ShowNodes,ELLIPSENODE_FORMAT)}
{ASKquery(ShowUseLinks,LINK_FORMAT)}
overlap=false
label=\"Program module use links \\\nExtracted from ConceptBase and layed out by Graphviz \"
fontsize=12;
\}
"
end


ELLIPSENODE_FORMAT in AnswerFormat with
pattern p: "node [shape=ellipse,style=filled,fillcolor=cornsilk,peripheries=1]; {this}\\n"
end


LINK_FORMAT in AnswerFormat with
pattern p: "{From({this})}->{To({this})} [color=black,len=1.0,label=\"\"];\\n"
end
