#!/bin/sh
# Syntax: genImports
# Author: Manfred Jeusfeld 
# Created: 2012-08-30 (2012-09-10)
# -----------------------------------------------------------------------
# Script for extracting import declarations from Prolog files using the
# pseudo-import clause of ConceptBase sources;
# Rather than doing the transformation, geImports only generates the 
# commands to perform the transformation. You need to pipe the output
# into CBshell and then to GraphViz to get the job done.
# You may want to use another layout algorithm such as fdp or circo
# instead of neato.
#
# Usage:
#   genImports [<srcdir>] [<query>] | cbshell -p | neato -Tpng > imports.png
#
# Examples
#  genImports /home/me/prj imports2GV  | cbshell -p | neato -Tpng > imports.png
#  genImports . uses2GV  | cbshell -p | neato -Tpng > uses.png
#  genImports  | cbshell -p | neato -Tpng > imports.png
#  genImports
# 
# When called without piping the output you will see the CBshell commands
# generated for processing the import declarations.
#

# author of the source code.
# Any re-distribution as source code must acknowledge the original author of this file.


# determine the absolute path of the directory containing genImports and ExtractImports.class
currdir=`pwd`
cd `dirname $0`
export swdir=`pwd`
cd $currdir

# if no source directory is provided as parameter of genImports, we take the current directory
srcdir=${1-$currdir}

# if $2 is defined that take it as mquery, otherwise take default query imports2GV
mquery=${2-imports2GV}


# generate the command to start a ConceptBase server 
echo "startServer -port 4201 -t no" 

# tell ConceptBase the required classes for storing import relations
echo tellModel ${swdir}/ImportsModel.sml.txt 


# generate the tell command for the import declarations
echo tell \" 

(
cd ${srcdir}
for file in *.pro
  do
    if [ -f "${file}" ]; then  
       java -cp ${swdir} ExtractImports "${file}"
    fi
done
)



echo \" 

# ask ConceptBase to produce the graph; the 'FRAME' answer format will be replaced by the
# right answer format for mquery; see attribute forQuery in ImportsModel.sml.txt
echo ask \"${mquery}\" OBJNAMES  FRAME  Now
echo showAnswer
