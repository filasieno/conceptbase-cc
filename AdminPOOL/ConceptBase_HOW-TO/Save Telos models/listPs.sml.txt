{*
* File P-Tuple.sml
* Author: Manfred Jeusfeld, jeusfeld@uvt.nl
* Date: 1-Oct-2009 (1-Oct-2009)
*----------------------------------------------------------------
* Prints the P-tuple of the current module.
* P-tuples are written as
*    P(id,source,label,destination)
*
*}



GenericQueryClass doListPs isA Proposition with
  parameter
     module: Module
   computed_attribute
     source: Proposition;
     label: Label;
     destination: Proposition
  constraint
    c: $ (~module contains ~this) and P(~this,~source,~label,~destination) $
end

AnswerFormat Pobject_Format with
   forQuery q: doListPs
   order o: ascending
   orderBy ob: "this.oid"
   pattern p:
"P({this.oid},{this.source.oid},{this.label},{this.destination.oid})
"
end


{* this is the main query; looks senseless but the work is done by listPLayout *}
listPs in QueryClass isA Boolean with
   constraint
     c1: $ (this=TRUE) $ 
end

AnswerFormat listPLayout with
  forQuery fq: listPs
  head hd: "\{* 
* Module: {currentmodule}.prop
* ======================================
* This is the list of P-objects (P-predicates) of {currentmodule}.
* Generated by:  {user} on ConceptBase {cb_version} ({cb_date_of_release})
* Transaction time: {transactiontime} (GMT) 
{IFTHENELSE({EQUAL({rollbacktime},Now)},*,* Rollback time: {rollbacktime} \(GMT\))}
* Original module path: {currentpath}
*\} 

"
  pattern pt:"{ASKquery(doListPs[{currentmodule}/module],Pobject_Format)}"
  tail t: "\{* end *\}"
end
