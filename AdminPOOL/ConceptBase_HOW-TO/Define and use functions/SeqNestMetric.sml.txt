* Example of evaluating hierachical metrics on parse trees of flowgraphs.
* The operators are sequencing and nesting. See also Fenton&Pfleeger: Software metrics.
* The recursive calculation of the metric requires the graph to be cycle-free.
* This works fine here because parse trees are by definition free of cycles.
* If the input graph would contain cycles, ConceptBase would not loop but rather
* return 'nil' (=undefined) as value of the metric.
* 
}

Construct with
  attribute
    part: Construct
end

{* flowgraph particles according to Fenton&Pfleeger: *}
P1 isA Construct end           {* simple statement *}
D0 isA Construct end           {* IF-THEN          *}
D1 isA Construct end           {* IF-THEN-ELSE     *}
D2 isA Construct end           {* WHILE-loop       *}
D3 isA Construct end           {* REPEAT-loop      *}

{* we only use the two-argument versions of Nest/Seq since
   Seq(F1;F2;F3) can be rewritten to Seq(F1;Seq(F2;F3)). Still,
   this is just to illustrate the principle.
*}

Seq isA Construct end          {* has two parts  *}
Nest isA Construct end         {* has two parts  *}


{* maxf returns the maximum of two integer numbers *}
Function maxf isA Integer with
  parameter
    x1: Integer;
    x2: Integer
  constraint
    ismax: $ (x1 > x2) and (this=x1) or
             not (x1 > x2) and (this=x2)   
           $
end


{* Hierarchical metric 'depth'                                *}
{* The variable 'this' holds the return value of the function *}

Function depth isA Integer with
  parameter
    x: Construct
  constraint 
     cval: $ (x in P1) and (this = 0) or
             (x in D0) and (this = 1) or
             (x in D1) and (this = 1) or
             (x in D2) and (this = 1) or
             (x in D3) and (this = 1) or
             (exists a1,a2/Construct (x in Seq) and (x part/p1 a1) and (x part/p2 a2) 
                       and (this = maxf(depth(a1),depth(a2))) ) or
             (exists b2/Construct (x in Nest) and (x part/p2 b2) and (this = depth(b2)+1) ) 
           $
end




{* A derived attribute like 'depth' can be used to display
   the metric values in the graph editor.
*}

Construct in Class with  
  attribute
    depth: Integer
  rule
    rval : $ forall n/Construct i/Integer (i = depth(n)) ==> (n depth i) $
end 




{* Example parse tree of a flowgraph
 1   IF A THEN S1 ENDIF;             ----> if1
 2   X=sqrt(...);                    ----> p11
 3   WHILE B DO                      ----> while1
 4     IF C THEN S0 ELSE S1 ENDIF;   ----> if2
 5   OD;

    ==> depth=2 due to IF-THEN-ELSE nested into WHILE
*}


seq1 in Seq with
  part
    p1: if1;
    p2: seq2
end

seq2 in Seq with
   part
    p1: p11;
    p2: nest1
end


if1 in D0 end
if2 in D1 end
while1 in D2 end
p11 in P1 end

nest1 in Nest with
  part 
    p1: while1;
    p2: if2
end
