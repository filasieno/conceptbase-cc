/* Some math functions functions */
/* We basically call the SWI-Prolog counterpart */

compute_sin(_res,_x,_C) :-
        cbserver:arg2val(_x,_v),
        number(_v),
        _vres is sin(_v),
        cbserver:val2arg(_vres,_res).

/* define the signature of 'sin' to ConceptBase */
tell: 'sin in Function isA Real with parameter x: Real end'.

compute_isInteger(_res,_x,_C) :-
        cbserver:arg2val(_x,_v),
        ((integer(_v),_vres='TRUE');
         _vres='FALSE'
        ),
        cbserver:makeId(_vres,_res).

tell: 'isInteger in Function isA Boolean with parameter x:Proposition comment c1: "TRUE if x is an integer number" end'.
