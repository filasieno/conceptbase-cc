compute_currentUser(_u) :-
   'CBserverInterface':active_user(_u),
   cbserver:makeId(_uname,_u),
   !.

tell: 'currentUser in Function isA CB_User end'.


compute_currentModule(_m) :-
   'GlobalPredicates':'M_SearchSpace'(_mod),
   cbserver:makeId(_mod,_m),
   !.

tell: 'currentModule in Function isA Module end'.
