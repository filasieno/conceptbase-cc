# Start: CBshell -f ECA-twoRules.cbs

# This script requires ConceptBase 7.2 or later.
# The script tests whether two rules that match the same event are
# executed in the order of their definition when no priorities are
# used. This should be true for all ECA modes.
#

startServer -u nonpersistent -port 4411 -t high


tell "

A in Class with
  attribute
    anext: A
end

B in Class with
  attribute
    bnext: B
end

C in Class isA A,B end

D in Class end

E in Class end

{* This query class ist just for triggerung ECArules; it uses YesClass as range, *}
{* so if you ask it from the user interface, you will get 'yes' as answer.       *}
Q1 in GenericQueryClass isA A with
  parameter
    p1: A
  constraint 
    c1: $ (p1 anext this) $
end

Q2 in GenericQueryClass isA B with
  parameter
    p1: B
  constraint 
    c1: $ (p1 bnext this) $
end
"



tell "

ECArule rule1 with
  mode m: ImmediateDeferred
  ecarule
        er : $  x/C a/A b/B
        ON Tell (x in C) 
        IF `(a in Q1[x]) and `(b in Q2[x])
        DO Tell (x in D) 
        ELSE Tell (x in E) 
        $
end

ECArule rule2 with
  mode m: ImmediateDeferred
  ecarule
        er : $  x/C a/A b/B
        ON Tell (x in C) 
        IF `(a in Q1[x]) and `(b in Q2[x])
        DO Tell (x in E) 
        ELSE Tell (x in D) 
        $
end

"


tell "
X1 in C with
  anext
   a1: A1
  bnext
   b1: B1
end

A1 in A end
B1 in B end
"


# The trace should show that rule1 is executed before rule2.
