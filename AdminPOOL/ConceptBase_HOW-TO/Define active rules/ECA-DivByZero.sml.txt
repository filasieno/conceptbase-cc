ECArule NoDivByZero with
  rejectMsg rm:
   "You should know that you cannot divide by zero!"
  ecarule
    r1: $ x/Integer
         ON Ask DIV[x/r1,0/r2]
         IF TRUE
         DO reject
        $
end
