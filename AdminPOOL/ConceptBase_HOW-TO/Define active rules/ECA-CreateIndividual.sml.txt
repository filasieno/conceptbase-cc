Class Manager with
attribute
    headOf : Department
end

Class Department end

ManagerWithoutDepartment in QueryClass isA Manager with
  constraint
     c1: $ not exists d/Department (this headOf d) $
end

Dep end  {* Prefix needs to be known *}


ECArule createDepartment with
mode m: Deferred
ecarule
        er : $ m/Manager d/Individual
        ON Tell (m in Manager)
        IF `(m in ManagerWithoutDepartment)
        DO CALL CreateIndividual(Dep,d),
           Tell (d in Department),
           Tell (m headOf/dep d)
        $
end
